// src/infrastructure/storage/MinIOClient.ts
import { Client as MinioClient, ClientOptions } from 'minio';
import { config } from '@/infrastructure/config';

let minioClient: MinioClient | null = null;

export function getMinioClient(): MinioClient {
  if (!minioClient) {
    const options: ClientOptions = {
      endPoint: config.storage.endPoint,
      port: config.storage.port,
      useSSL: config.storage.useSSL,
      accessKey: config.storage.accessKey,
      secretKey: config.storage.secretKey,
    };
    minioClient = new MinioClient(options);
  }
  return minioClient;
}

export async function ensureBucketExists(bucketName: string): Promise<void> {
  const client = getMinioClient();
  const exists = await client.bucketExists(bucketName);
  if (!exists) {
    await client.makeBucket(bucketName, 'us-east-1');
    console.log(`📦 Created bucket: ${bucketName}`);
  }
}

export async function initializeStorage(): Promise<void> {
  await ensureBucketExists(config.storage.mediaBucket);
  console.log('📦 MinIO storage initialized');
}
```

import { ValidationError } from '@/domain/errors';

export type MediaType = 'PHOTO' | 'VIDEO' | 'DOCUMENT' | 'AUDIO';

interface MediaProps {
  id?: string;
  familyId: string;
  type: MediaType;
  storageKey: string;
  originalName: string;
  mimeType: string;
  size: number;
  width?: number;
  height?: number;
  duration?: number;
  uploadedBy: string;
  description?: string;
  tags?: string[];
  createdAt?: Date;
  updatedAt?: Date;
}

export class Media {
  readonly id?: string;
  readonly familyId: string;
  readonly type: MediaType;
  readonly storageKey: string;
  readonly originalName: string;
  readonly mimeType: string;
  readonly size: number;
  readonly width?: number;
  readonly height?: number;
  readonly duration?: number;
  readonly uploadedBy: string;
  readonly description?: string;
  readonly tags: string[];
  readonly createdAt: Date;
  readonly updatedAt: Date;

  private constructor(props: MediaProps) {
    this.id = props.id;
    this.familyId = props.familyId;
    this.type = props.type;
    this.storageKey = props.storageKey;
    this.originalName = props.originalName;
    this.mimeType = props.mimeType;
    this.size = props.size;
    this.width = props.width;
    this.height = props.height;
    this.duration = props.duration;
    this.uploadedBy = props.uploadedBy;
    this.description = props.description;
    this.tags = props.tags ?? [];
    this.createdAt = props.createdAt ?? new Date();
    this.updatedAt = props.updatedAt ?? new Date();
  }

  static create(props: MediaProps): Media {
    if (!props.storageKey?.trim()) {
      throw new ValidationError('Storage key is required');
    }
    if (!props.originalName?.trim()) {
      throw new ValidationError('Original name is required');
    }
    if (props.size <= 0) {
      throw new ValidationError('Size must be positive');
    }
    return new Media(props);
  }

  isImage(): boolean {
    return this.type === 'PHOTO';
  }

  isVideo(): boolean {
    return this.type === 'VIDEO';
  }

  getFileExtension(): string {
    return this.originalName.split('.').pop() ?? '';
  }

  toJSON() {
    return {
      id: this.id,
      familyId: this.familyId,
      type: this.type,
      storageKey: this.storageKey,
      originalName: this.originalName,
      mimeType: this.mimeType,
      size: this.size,
      width: this.width,
      height: this.height,
      duration: this.duration,
      uploadedBy: this.uploadedBy,
      description: this.description,
      tags: this.tags,
      createdAt: this.createdAt,
      updatedAt: this.updatedAt,
    };
  }
}
```

import mongoose, { Schema, Document } from 'mongoose';

export type MediaType = 'PHOTO' | 'VIDEO' | 'DOCUMENT' | 'AUDIO';

export interface MediaDocument extends Document {
  familyId: mongoose.Types.ObjectId;
  type: MediaType;
  storageKey: string;
  originalName: string;
  mimeType: string;
  size: number;
  width?: number;
  height?: number;
  duration?: number;
  uploadedBy: mongoose.Types.ObjectId;
  description?: string;
  tags: string[];
  createdAt: Date;
  updatedAt: Date;
}

const MediaSchema = new Schema<MediaDocument>({
  familyId: { type: Schema.Types.ObjectId, ref: 'Family', required: true, index: true },
  type: { type: String, enum: ['PHOTO', 'VIDEO', 'DOCUMENT', 'AUDIO'], required: true },
  storageKey: { type: String, required: true, unique: true },
  originalName: { type: String, required: true, maxlength: 255 },
  mimeType: { type: String, required: true, maxlength: 100 },
  size: { type: Number, required: true, min: 0 },
  width: { type: Number, min: 0 },
  height: { type: Number, min: 0 },
  duration: { type: Number, min: 0 },
  uploadedBy: { type: Schema.Types.ObjectId, ref: 'User', required: true },
  description: { type: String, maxlength: 1000 },
  tags: [{ type: String, maxlength: 50 }],
}, { timestamps: true });

// Index for searching by tags
MediaSchema.index({ tags: 1 });
MediaSchema.index({ familyId: 1, type: 1 });

export const MediaModel = mongoose.model<MediaDocument>('Media', MediaSchema);

// src/infrastructure/database/repositories/MediaRepository.ts
import { Media, MediaType } from '@/domain/entities';
import { IMediaRepository, CreateMediaInput, MediaFilter } from '@/domain/interfaces';
import { MediaModel } from '../models';
import { NotFoundError } from '@/domain/errors';

export class MediaRepository implements IMediaRepository {
  async findById(id: string): Promise<Media | null> {
    const doc = await MediaModel.findById(id);
    return doc ? this.toEntity(doc) : null;
  }

  async findByFamilyId(familyId: string, filter?: MediaFilter): Promise<Media[]> {
    let query: any = { familyId };

    if (filter?.type) {
      query.type = filter.type;
    }
    if (filter?.tags && filter.tags.length > 0) {
      query.tags = { $in: filter.tags };
    }

    const docs = await MediaModel.find(query).sort({ createdAt: -1 });
    return docs.map(doc => this.toEntity(doc));
  }

  async create(input: CreateMediaInput): Promise<Media> {
    const doc = await MediaModel.create(input);
    return this.toEntity(doc);
  }

  async update(id: string, updates: Partial<Media>): Promise<Media> {
    const doc = await MediaModel.findByIdAndUpdate(
      id,
      { ...updates, updatedAt: new Date() },
      { new: true, runValidators: true }
    );
    if (!doc) throw new NotFoundError('Media', id);
    return this.toEntity(doc);
  }

  async delete(id: string): Promise<void> {
    await MediaModel.findByIdAndDelete(id);
  }

  async findByStorageKey(storageKey: string): Promise<Media | null> {
    const doc = await MediaModel.findOne({ storageKey });
    return doc ? this.toEntity(doc) : null;
  }

  private toEntity(doc: any): Media {
    return Media.create({
      id: doc._id.toString(),
      familyId: doc.familyId.toString(),
      type: doc.type as MediaType,
      storageKey: doc.storageKey,
      originalName: doc.originalName,
      mimeType: doc.mimeType,
      size: doc.size,
      width: doc.width,
      height: doc.height,
      duration: doc.duration,
      uploadedBy: doc.uploadedBy.toString(),
      description: doc.description,
      tags: doc.tags,
      createdAt: doc.createdAt,
      updatedAt: doc.updatedAt,
    });
  }
}

// src/application/services/MediaService.ts
import { IMediaRepository, IUserRepository } from '@/domain/interfaces';
import { UnauthorizedError, NotFoundError, ValidationError } from '@/domain/errors';
import { Media, MediaType } from '@/domain/entities';
import { getMinioClient } from '@/infrastructure/storage/MinIOClient';
import { config } from '@/infrastructure/config';
import { Readable } from 'stream';

const ALLOWED_MIME_TYPES: Record<MediaType, string[]> = {
  PHOTO: ['image/jpeg', 'image/png', 'image/gif', 'image/webp'],
  VIDEO: ['video/mp4', 'video/webm', 'video/quicktime'],
  DOCUMENT: ['application/pdf', 'application/msword', 'text/plain'],
  AUDIO: ['audio/mpeg', 'audio/wav', 'audio/ogg'],
};

const MAX_FILE_SIZE = 50 * 1024 * 1024; // 50MB

export interface UploadInput {
  file: {
    data: Buffer;
    mimetype: string;
    originalname: string;
    size: number;
  };
  type: MediaType;
  description?: string;
  tags?: string[];
}

export class MediaService {
  constructor(
    private mediaRepo: IMediaRepository,
    private userRepo: IUserRepository,
  ) {}

  async uploadMedia(userId: string, familyId: string, input: UploadInput): Promise<Media> {
    const user = await this.userRepo.findById(userId);
    if (!user || !user.canEdit()) {
      throw new UnauthorizedError('Cannot upload media');
    }
    if (user.familyId !== familyId) {
      throw new UnauthorizedError('Cannot upload to other family');
    }

    // Validate file
    if (input.file.size > MAX_FILE_SIZE) {
      throw new ValidationError('File size exceeds 50MB limit');
    }
    const allowedTypes = ALLOWED_MIME_TYPES[input.type];
    if (!allowedTypes.includes(input.file.mimetype)) {
      throw new ValidationError(`Invalid mime type for ${input.type}`);
    }

    // Generate storage key
    const timestamp = Date.now();
    const extension = input.file.originalname.split('.').pop();
    const storageKey = `media/${familyId}/${timestamp}-${Math.random().toString(36).slice(2)}.${extension}`;

    // Upload to MinIO
    const minioClient = getMinioClient();
    await minioClient.putObject(
      config.storage.mediaBucket,
      storageKey,
      input.file.data,
      input.file.size,
      { 'Content-Type': input.file.mimetype }
    );

    // Create media record
    const media = await this.mediaRepo.create({
      familyId,
      type: input.type,
      storageKey,
      originalName: input.file.originalname,
      mimeType: input.file.mimetype,
      size: input.file.size,
      uploadedBy: userId,
      description: input.description,
      tags: input.tags,
    });

    return media;
  }

  async getMediaUrl(userId: string, mediaId: string): Promise<string> {
    const user = await this.userRepo.findById(userId);
    if (!user) {
      throw new UnauthorizedError('User not found');
    }

    const media = await this.mediaRepo.findById(mediaId);
    if (!media) {
      throw new NotFoundError('Media', mediaId);
    }
    if (media.familyId !== user.familyId) {
      throw new UnauthorizedError('Cannot access media from other family');
    }

    const minioClient = getMinioClient();
    const url = await minioClient.presignedGetObject(
      config.storage.mediaBucket,
      media.storageKey,
      60 * 60 // 1 hour expiry
    );
    return url;
  }

  async deleteMedia(userId: string, mediaId: string): Promise<void> {
    const user = await this.userRepo.findById(userId);
    if (!user || !user.canEdit()) {
      throw new UnauthorizedError('Cannot delete media');
    }

    const media = await this.mediaRepo.findById(mediaId);
    if (!media) {
      throw new NotFoundError('Media', mediaId);
    }
    if (media.familyId !== user.familyId) {
      throw new UnauthorizedError('Cannot delete media from other family');
    }

    // Delete from MinIO
    const minioClient = getMinioClient();
    await minioClient.removeObject(config.storage.mediaBucket, media.storageKey);

    // Delete from database
    await this.mediaRepo.delete(mediaId);
  }

  async getFamilyMedia(userId: string, familyId: string, filter?: any): Promise<Media[]> {
    const user = await this.userRepo.findById(userId);
    if (!user) {
      throw new UnauthorizedError('User not found');
    }
    if (user.familyId !== familyId) {
      throw new UnauthorizedError('Cannot access media from other family');
    }

    return await this.mediaRepo.findByFamilyId(familyId, filter);
  }
}
