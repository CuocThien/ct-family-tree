```typescript
// src/domain/entities/Member.ts
import { ValidationError } from '@/domain/errors';
import { LifeEvent } from '@/domain/value-objects';

interface PersonalInfo {
  firstName: string;
  lastName: string;
  middleName?: string;
  maidenName?: string;
  gender?: 'MALE' | 'FEMALE' | 'OTHER';
  photoId?: string;
}

interface Education {
  institution: string;
  degree?: string;
  field?: string;
  startDate?: { year: number; month?: number };
  endDate?: { year: number; month?: number };
}

interface ContactInfo {
  email?: string;
  phone?: string;
  address?: { city?: string; country: string };
}

interface MedicalInfo {
  bloodType?: string;
  allergies?: string[];
  conditions?: string[];
  notes?: string;
}

interface MemberProps {
  id?: string;
  familyId: string;
  personalInfo: PersonalInfo;
  lifeEvents: LifeEvent[];
  bio?: string;
  occupation?: string;
  education: Education[];
  contact?: ContactInfo;
  medicalHistory?: MedicalInfo;
  createdAt?: Date;
  updatedAt?: Date;
}

export class Member {
  readonly id?: string;
  readonly familyId: string;
  readonly personalInfo: PersonalInfo;
  readonly lifeEvents: LifeEvent[];
  readonly bio?: string;
  readonly occupation?: string;
  readonly education: Education[];
  readonly contact?: ContactInfo;
  readonly medicalHistory?: MedicalInfo;
  readonly createdAt: Date;
  readonly updatedAt: Date;

  private constructor(props: MemberProps) {
    this.id = props.id;
    this.familyId = props.familyId;
    this.personalInfo = props.personalInfo;
    this.lifeEvents = props.lifeEvents ?? [];
    this.bio = props.bio;
    this.occupation = props.occupation;
    this.education = props.education ?? [];
    this.contact = props.contact;
    this.medicalHistory = props.medicalHistory;
    this.createdAt = props.createdAt ?? new Date();
    this.updatedAt = props.updatedAt ?? new Date();
  }

  static create(props: MemberProps): Member {
    if (!props.personalInfo.firstName?.trim()) {
      throw new ValidationError('First name is required');
    }
    if (!props.personalInfo.lastName?.trim()) {
      throw new ValidationError('Last name is required');
    }
    return new Member(props);
  }

  hasBirthEvent(): boolean {
    return this.lifeEvents.some(e => e.type === 'BIRTH');
  }

  hasDeathEvent(): boolean {
    return this.lifeEvents.some(e => e.type === 'DEATH');
  }

  isLiving(): boolean {
    return !this.hasDeathEvent();
  }

  getFullName(): string {
    const parts = [this.personalInfo.firstName, this.personalInfo.middleName, this.personalInfo.lastName];
    return parts.filter(Boolean).join(' ');
  }

  toJSON() {
    return {
      id: this.id,
      familyId: this.familyId,
      personalInfo: this.personalInfo,
      lifeEvents: this.lifeEvents.map(e => e.toJSON()),
      bio: this.bio,
      occupation: this.occupation,
      education: this.education,
      contact: this.contact,
      medicalHistory: this.medicalHistory,
      createdAt: this.createdAt,
      updatedAt: this.updatedAt,
    };
  }
}
```

```typescript
// src/infrastructure/database/models/Member.ts
import mongoose, { Schema, Document } from 'mongoose';

interface PersonalInfo {
  firstName: string;
  lastName: string;
  middleName?: string;
  maidenName?: string;
  gender?: 'MALE' | 'FEMALE' | 'OTHER';
  photoId?: mongoose.Types.ObjectId;
}

interface LifeEvent {
  type: 'BIRTH' | 'DEATH' | 'MARRIAGE' | 'WORK' | 'CUSTOM';
  customType?: string;
  date?: { year: number; month?: number; day?: number; isApproximate?: boolean };
  location?: { street?: string; city?: string; state?: string; country: string };
  notes?: string;
  mediaIds?: mongoose.Types.ObjectId[];
}

interface Education {
  institution: string;
  degree?: string;
  field?: string;
  startDate?: { year: number; month?: number };
  endDate?: { year: number; month?: number };
}

export interface MemberDocument extends Document {
  familyId: mongoose.Types.ObjectId;
  personalInfo: PersonalInfo;
  lifeEvents: LifeEvent[];
  bio?: string;
  occupation?: string;
  education: Education[];
  contact?: {
    email?: string;
    phone?: string;
    address?: { street?: string; city?: string; state?: string; country: string };
  };
  medicalHistory?: {
    bloodType?: string;
    allergies?: string[];
    conditions?: string[];
    notes?: string;
  };
  createdAt: Date;
  updatedAt: Date;
}

const PersonalInfoSchema = new Schema({
  firstName: { type: String, required: true, maxlength: 100 },
  lastName: { type: String, required: true, maxlength: 100 },
  middleName: { type: String, maxlength: 100 },
  maidenName: { type: String, maxlength: 100 },
  gender: { type: String, enum: ['MALE', 'FEMALE', 'OTHER'] },
  photoId: { type: Schema.Types.ObjectId, ref: 'Media' },
}, { _id: false });

const LifeEventSchema = new Schema({
  type: { type: String, enum: ['BIRTH', 'DEATH', 'MARRIAGE', 'WORK', 'CUSTOM'], required: true },
  customType: { type: String, maxlength: 50 },
  date: {
    year: { type: Number, min: 1, max: 2100 },
    month: { type: Number, min: 1, max: 12 },
    day: { type: Number, min: 1, max: 31 },
    isApproximate: { type: Boolean, default: false },
  },
  location: {
    street: { type: String, maxlength: 200 },
    city: { type: String, maxlength: 100 },
    state: { type: String, maxlength: 100 },
    country: { type: String, maxlength: 100 },
  },
  notes: { type: String, maxlength: 2000 },
  mediaIds: [{ type: Schema.Types.ObjectId, ref: 'Media' }],
}, { _id: false });

const EducationSchema = new Schema({
  institution: { type: String, required: true, maxlength: 200 },
  degree: { type: String, maxlength: 100 },
  field: { type: String, maxlength: 100 },
  startDate: { year: Number, month: { type: Number, min: 1, max: 12 } },
  endDate: { year: Number, month: { type: Number, min: 1, max: 12 } },
}, { _id: false });

const MemberSchema = new Schema<MemberDocument>({
  familyId: { type: Schema.Types.ObjectId, ref: 'Family', required: true, index: true },
  personalInfo: { type: PersonalInfoSchema, required: true },
  lifeEvents: [LifeEventSchema],
  bio: { type: String, maxlength: 5000 },
  occupation: { type: String, maxlength: 200 },
  education: [EducationSchema],
  contact: {
    email: { type: String, maxlength: 255 },
    phone: { type: String, maxlength: 50 },
    address: {
      street: { type: String, maxlength: 200 },
      city: { type: String, maxlength: 100 },
      state: { type: String, maxlength: 100 },
      country: { type: String, maxlength: 100 },
    },
  },
  medicalHistory: {
    bloodType: { type: String, maxlength: 10 },
    allergies: [{ type: String, maxlength: 100 }],
    conditions: [{ type: String, maxlength: 100 }],
    notes: { type: String, maxlength: 2000 },
  },
}, { timestamps: true });

// Text search index
MemberSchema.index(
  {
    'personalInfo.firstName': 'text',
    'personalInfo.lastName': 'text',
    bio: 'text',
    occupation: 'text',
  },
  { weights: { 'personalInfo.firstName': 10, 'personalInfo.lastName': 10, bio: 5, occupation: 3 } }
);

export const MemberModel = mongoose.model<MemberDocument>('Member', MemberSchema);
```

### Repository Implementation
```typescript
// src/infrastructure/database/repositories/MemberRepository.ts
import { Member } from '@/domain/entities';
import { IMemberRepository, CreateMemberInput, MemberFilter } from '@/domain/interfaces';
import { MemberModel } from '../models';
import { NotFoundError } from '@/domain/errors';

export class MemberRepository implements IMemberRepository {
  async findById(id: string): Promise<Member | null> {
    const doc = await MemberModel.findById(id).populate('personalInfo.photoId');
    return doc ? this.toEntity(doc) : null;
  }

  async findByFamilyId(familyId: string, filter?: MemberFilter): Promise<Member[]> {
    let query: any = { familyId };

    if (filter?.gender) {
      query['personalInfo.gender'] = filter.gender;
    }

    let docs = await MemberModel.find(query).sort({ 'personalInfo.lastName': 1, 'personalInfo.firstName': 1 });

    // Filter by living status (in-memory since it depends on lifeEvents)
    if (filter?.isLiving !== undefined) {
      docs = docs.filter(doc => {
        const hasDeath = doc.lifeEvents.some(e => e.type === 'DEATH');
        return filter.isLiving ? !hasDeath : hasDeath;
      });
    }

    return docs.map(doc => this.toEntity(doc));
  }

  async create(input: CreateMemberInput): Promise<Member> {
    const doc = await MemberModel.create(input);
    return this.toEntity(doc);
  }

  async update(id: string, updates: Partial<Member>): Promise<Member> {
    const doc = await MemberModel.findByIdAndUpdate(
      id,
      { ...updates, updatedAt: new Date() },
      { new: true, runValidators: true }
    );
    if (!doc) throw new NotFoundError('Member', id);
    return this.toEntity(doc);
  }

  async delete(id: string): Promise<void> {
    await MemberModel.findByIdAndDelete(id);
  }

  async search(familyId: string, query: string, options?: { limit?: number; skip?: number }): Promise<{ members: Member[]; totalCount: number }> {
    const limit = options?.limit ?? 20;
    const skip = options?.skip ?? 0;

    const searchQuery = {
      familyId,
      $text: { $search: query },
    };

    const [docs, totalCount] = await Promise.all([
      MemberModel.find(searchQuery)
        .sort({ score: { $meta: 'textScore' } })
        .skip(skip)
        .limit(limit),
      MemberModel.countDocuments(searchQuery),
    ]);

    return {
      members: docs.map(doc => this.toEntity(doc)),
      totalCount,
    };
  }

  private toEntity(doc: any): Member {
    return Member.create({
      id: doc._id.toString(),
      familyId: doc.familyId.toString(),
      personalInfo: {
        firstName: doc.personalInfo.firstName,
        lastName: doc.personalInfo.lastName,
        middleName: doc.personalInfo.middleName,
        maidenName: doc.personalInfo.maidenName,
        gender: doc.personalInfo.gender,
        photoId: doc.personalInfo.photoId?.toString(),
      },
      lifeEvents: doc.lifeEvents.map((e: any) => ({
        type: e.type,
        customType: e.customType,
        date: e.date,
        location: e.location,
        notes: e.notes,
        mediaIds: e.mediaIds?.map((id: any) => id.toString()),
      })),
      bio: doc.bio,
      occupation: doc.occupation,
      education: doc.education,
      contact: doc.contact,
      medicalHistory: doc.medicalHistory,
      createdAt: doc.createdAt,
      updatedAt: doc.updatedAt,
    });
  }
}
```

```typescript
// src/application/services/MemberService.ts
import { IMemberRepository, IFamilyRepository, IUserRepository } from '@/domain/interfaces';
import { UnauthorizedError, NotFoundError, ValidationError } from '@/domain/errors';
import { Member } from '@/domain/entities';
import { z } from 'zod';

const createMemberSchema = z.object({
  familyId: z.string(),
  personalInfo: z.object({
    firstName: z.string().min(1).max(100),
    lastName: z.string().min(1).max(100),
    middleName: z.string().max(100).optional(),
    maidenName: z.string().max(100).optional(),
    gender: z.enum(['MALE', 'FEMALE', 'OTHER']).optional(),
    photoId: z.string().optional(),
  }),
  bio: z.string().max(5000).optional(),
  occupation: z.string().max(200).optional(),
});

export class MemberService {
  constructor(
    private memberRepo: IMemberRepository,
    private familyRepo: IFamilyRepository,
    private userRepo: IUserRepository,
  ) {}

  async createMember(userId: string, input: z.infer<typeof createMemberSchema>): Promise<Member> {
    const user = await this.userRepo.findById(userId);
    if (!user || !user.canEdit()) {
      throw new UnauthorizedError('Cannot create members');
    }

    const validated = createMemberSchema.parse(input);

    // Verify family exists and user belongs to it
    const family = await this.familyRepo.findById(validated.familyId);
    if (!family) {
      throw new NotFoundError('Family', validated.familyId);
    }
    if (user.familyId !== validated.familyId) {
      throw new UnauthorizedError('Cannot create members in other families');
    }

    // Check for duplicate birth events
    if (input.personalInfo.firstName && input.personalInfo.lastName) {
      // Could add duplicate checking logic here
    }

    return await this.memberRepo.create(validated);
  }

  async updateMember(userId: string, memberId: string, updates: Partial<Member>): Promise<Member> {
    const user = await this.userRepo.findById(userId);
    if (!user || !user.canEdit()) {
      throw new UnauthorizedError('Cannot update members');
    }

    const member = await this.memberRepo.findById(memberId);
    if (!member) {
      throw new NotFoundError('Member', memberId);
    }
    if (member.familyId !== user.familyId) {
      throw new UnauthorizedError('Cannot update members from other families');
    }

    return await this.memberRepo.update(memberId, updates);
  }

  async deleteMember(userId: string, memberId: string): Promise<void> {
    const user = await this.userRepo.findById(userId);
    if (!user || !user.canEdit()) {
      throw new UnauthorizedError('Cannot delete members');
    }

    const member = await this.memberRepo.findById(memberId);
    if (!member) {
      throw new NotFoundError('Member', memberId);
    }
    if (member.familyId !== user.familyId) {
      throw new UnauthorizedError('Cannot delete members from other families');
    }

    await this.memberRepo.delete(memberId);
  }

  async getMembers(userId: string, familyId: string, filter?: any): Promise<Member[]> {
    const user = await this.userRepo.findById(userId);
    if (!user) {
      throw new UnauthorizedError('User not found');
    }
    if (user.familyId !== familyId) {
      throw new UnauthorizedError('Cannot view members from other families');
    }

    return await this.memberRepo.findByFamilyId(familyId, filter);
  }

  async getMember(userId: string, memberId: string): Promise<Member> {
    const user = await this.userRepo.findById(userId);
    if (!user) {
      throw new UnauthorizedError('User not found');
    }

    const member = await this.memberRepo.findById(memberId);
    if (!member) {
      throw new NotFoundError('Member', memberId);
    }
    if (member.familyId !== user.familyId) {
      throw new UnauthorizedError('Cannot view members from other families');
    }

    return member;
  }

  async searchMembers(userId: string, query: string, options?: { limit?: number }): Promise<{ members: Member[]; totalCount: number }> {
    const user = await this.userRepo.findById(userId);
    if (!user) {
      throw new UnauthorizedError('User not found');
    }

    return await this.memberRepo.search(user.familyId, query, options);
  }
}
```

```typescript
import { ValidationError } from '@/domain/errors';

export type RelationshipType = 'PARENT' | 'SPOUSE' | 'PARTNER' | 'ADOPTION';

interface RelationshipProps {
  id?: string;
  familyId: string;
  type: RelationshipType;
  person1Id: string;
  person2Id: string;
  startDate?: Date;
  endDate?: Date;
  isCurrent: boolean;
  metadata?: Record<string, any>;
  createdAt?: Date;
  updatedAt?: Date;
}

export class Relationship {
  readonly id?: string;
  readonly familyId: string;
  readonly type: RelationshipType;
  readonly person1Id: string;
  readonly person2Id: string;
  readonly startDate?: Date;
  readonly endDate?: Date;
  readonly isCurrent: boolean;
  readonly metadata?: Record<string, any>;
  readonly createdAt: Date;
  readonly updatedAt: Date;

  private constructor(props: RelationshipProps) {
    this.id = props.id;
    this.familyId = props.familyId;
    this.type = props.type;
    this.person1Id = props.person1Id;
    this.person2Id = props.person2Id;
    this.startDate = props.startDate;
    this.endDate = props.endDate;
    this.isCurrent = props.isCurrent;
    this.metadata = props.metadata;
    this.createdAt = props.createdAt ?? new Date();
    this.updatedAt = props.updatedAt ?? new Date();
  }

  static create(props: RelationshipProps): Relationship {
    if (props.person1Id === props.person2Id) {
      throw new ValidationError('Cannot create self-relationship');
    }
    const validTypes: RelationshipType[] = ['PARENT', 'SPOUSE', 'PARTNER', 'ADOPTION'];
    if (!validTypes.includes(props.type)) {
      throw new ValidationError('Invalid relationship type');
    }
    return new Relationship(props);
  }

  involves(memberId: string): boolean {
    return this.person1Id === memberId || this.person2Id === memberId;
  }

  getOtherMemberId(memberId: string): string {
    if (this.person1Id === memberId) return this.person2Id;
    if (this.person2Id === memberId) return this.person1Id;
    throw new Error('Member not involved in this relationship');
  }

  toJSON() {
    return {
      id: this.id,
      familyId: this.familyId,
      type: this.type,
      person1Id: this.person1Id,
      person2Id: this.person2Id,
      startDate: this.startDate,
      endDate: this.endDate,
      isCurrent: this.isCurrent,
      metadata: this.metadata,
      createdAt: this.createdAt,
      updatedAt: this.updatedAt,
    };
  }
}
```
