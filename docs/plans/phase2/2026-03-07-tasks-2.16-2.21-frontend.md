```typescript
// src/lib/apollo/client.ts
import { ApolloClient, ApolloLink, HttpLink, from } from '@apollo/client';
import { onError } from '@apollo/client/link/error';
import { setContext } from '@apollo/client/link/context';
import { cache, isLoggedInVar } from './cache';

const httpLink = new HttpLink({
  uri: import.meta.env.VITE_API_URL || 'http://localhost:3000/graphql',
});

const authLink = setContext((_, { headers }) => {
  const token = localStorage.getItem('token');
  return {
    headers: {
      ...headers,
      authorization: token ? `Bearer ${token}` : '',
    },
  };
});

const errorLink = onError(({ graphQLErrors, networkError }) => {
  if (graphQLErrors) {
    for (const err of graphQLErrors) {
      if (err.extensions?.code === 'UNAUTHORIZED') {
        localStorage.removeItem('token');
        isLoggedInVar(false);
        window.location.href = '/login';
      }
    }
  }
  if (networkError) {
    console.error('Network error:', networkError);
  }
});

export const apolloClient = new ApolloClient({
  link: from([errorLink, authLink, httpLink]),
  cache,
  defaultOptions: {
    watchQuery: {
      fetchPolicy: 'cache-and-network',
    },
  },
});
```

```typescript
import { atom, atomWithStorage } from 'jotai';
import { User } from '@/types/graphql';

// Persistent atoms
export const tokenAtom = atomWithStorage<string | null>('token', null);

// Derived atoms
export const isAuthenticatedAtom = atom((get) => {
  const token = get(tokenAtom);
  return !!token;
});

export const currentUserAtom = atom<User | null>(null);

export const currentFamilyIdAtom = atom((get) => {
  const user = get(currentUserAtom);
  return user?.familyId ?? null;
});

export const userRoleAtom = atom((get) => {
  const user = get(currentUserAtom);
  return user?.role ?? null;
});

export const canEditAtom = atom((get) => {
  const role = get(userRoleAtom);
  return role === 'OWNER' || role === 'EDITOR';
});
```

```typescript
// src/stores/auth/hooks.ts
import { useAtom, useAtomValue, useSetAtom } from 'jotai';
import { useCallback } from 'react';
import { useNavigate } from 'react-router-dom';
import {
  tokenAtom,
  isAuthenticatedAtom,
  currentUserAtom,
  canEditAtom,
} from './atoms';

export function useAuth() {
  const [token, setToken] = useAtom(tokenAtom);
  const [user, setUser] = useAtom(currentUserAtom);
  const isAuthenticated = useAtomValue(isAuthenticatedAtom);
  const canEdit = useAtomValue(canEditAtom);
  const navigate = useNavigate();

  const login = useCallback((newToken: string, userData: any) => {
    setToken(newToken);
    setUser(userData);
  }, [setToken, setUser]);

  const logout = useCallback(() => {
    setToken(null);
    setUser(null);
    navigate('/login');
  }, [setToken, setUser, navigate]);

  return {
    token,
    user,
    isAuthenticated,
    canEdit,
    login,
    logout,
  };
}
```

```typescript
import { Card, Avatar, Typography, Tag } from 'antd';
import { UserOutlined } from '@ant-design/icons';
import { Link } from 'react-router-dom';

const { Text, Title } = Typography;

interface MemberCardProps {
  member: {
    id: string;
    personalInfo: {
      firstName: string;
      lastName: string;
      middleName?: string;
      gender?: string;
    };
    lifeEvents: Array<{ type: string; date?: { year: number } }>;
    occupation?: string;
  };
}

export function MemberCard({ member }: MemberCardProps) {
  const birthEvent = member.lifeEvents.find(e => e.type === 'BIRTH');
  const deathEvent = member.lifeEvents.find(e => e.type === 'DEATH');
  const isLiving = !deathEvent;

  const fullName = [
    member.personalInfo.firstName,
    member.personalInfo.middleName,
    member.personalInfo.lastName,
  ].filter(Boolean).join(' ');

  const lifeSpan = birthEvent?.date?.year
    ? `${birthEvent.date.year}${deathEvent?.date?.year ? ` - ${deathEvent.date.year}` : isLiving ? ' - Present' : ''}`
    : null;

  return (
    <Link to={`/members/${member.id}`}>
      <Card
        hoverable
        className="mb-4"
        cover={
          <div className="h-32 bg-gray-100 flex items-center justify-center">

          <Text type="secondary" className="block mb-2">{member.occupation}</Text>
        )}
        {lifeSpan && (
          <Text type="secondary" className="block">{lifeSpan}</Text>
        )}
```

```typescript
// src/features/members/pages/MemberListPage.tsx
import { useState } from 'react';
import { Row, Col, Input, Select, Spin, Empty } from 'antd';
import { useQuery } from '@apollo/client';
import { MemberCard } from '../components/MemberCard';
import { GET_MEMBERS } from '../graphql/queries';
import { useAuth } from '@/stores/auth';

const { Search } = Input;
const { Option } = Select;

export function MemberListPage() {
  const { user } = useAuth();
  const [searchQuery, setSearchQuery] = useState('');
  const [genderFilter, setGenderFilter] = useState<string | undefined>();

  const { loading, error, data } = useQuery(GET_MEMBERS, {
    variables: {
      familyId: user?.familyId,
      filter: { gender: genderFilter },
    },
    skip: !user?.familyId,
  });

  if (loading) {
    return (
      <div className="flex justify-center items-center h-64">
        <Spin size="large" />
      </div>
    );
  }

  if (error) {
    return <div>Error: {error.message}</div>;
  }

  const members = data?.members || [];

  // Client-side search filter
  const filteredMembers = searchQuery
    ? members.filter((m: any) => {
        const name = `${m.personalInfo.firstName} ${m.personalInfo.lastName}`.toLowerCase();
        return name.includes(searchQuery.toLowerCase());
      })
    : members;

  return (
    <div className="p-6">
      <h1 className="text-2xl font-bold mb-6">Family Members</h1>

      <div className="flex gap-4 mb-6">
        <Search
          placeholder="Search members..."
          allowClear
          onChange={(e) => setSearchQuery(e.target.value)}
          style={{ width: 300 }}
        />
        <Select
          placeholder="Filter by gender"
          allowClear
          style={{ width: 150 }}
          onChange={setGenderFilter}
        >
          <Option value="MALE">Male</Option>
          <Option value="FEMALE">Female</Option>
          <Option value="OTHER">Other</Option>
        </Select>
      </div>

      {filteredMembers.length === 0 ? (
        <Empty description="No members found" />
      ) : (
        <Row gutter={[16, 16]}>
          {filteredMembers.map((member: any) => (
            <Col xs={24} sm={12} md={8} lg={6} key={member.id}>
              <MemberCard member={member} />
            </Col>
          ))}
        </Row>
      )}
    </div>
  );
}
```

```typescript
import { useNavigate } from 'react-router-dom';

const { TextArea } = Input;
const { Option } = Select;

interface MemberFormProps {
```

```typescript
// src/features/tree/utils/layout.ts
interface TreeNode {
  id: string;
  name: string;
  gender?: string;
  x: number;
  y: number;
  generation: number;
  spouseIds: string[];
  parentIds: string[];
  childrenIds: string[];
}

interface TreeLink {
  source: string;
  target: string;
  type: 'parent' | 'spouse';
}

const NODE_WIDTH = 180;
const NODE_HEIGHT = 80;
const HORIZONTAL_SPACING = 40;
const VERTICAL_SPACING = 100;

export function calculateTreeLayout(
  members: any[],
  relationships: any[]
): { nodes: TreeNode[]; links: TreeLink[] } {
  const nodeMap = new Map<string, TreeNode>();
  const links: TreeLink[] = [];

  // Create initial nodes
  members.forEach((member, index) => {
    nodeMap.set(member.id, {
      id: member.id,
      name: `${member.personalInfo.firstName} ${member.personalInfo.lastName}`,
      gender: member.personalInfo.gender,
      x: 0,
      y: 0,
      generation: 0,
      spouseIds: [],
      parentIds: [],
      childrenIds: [],
    });
  });

  // Process relationships
  relationships.forEach((rel) => {
    const node1 = nodeMap.get(rel.person1Id);
    const node2 = nodeMap.get(rel.person2Id);

    if (rel.type === 'PARENT') {
      // person1 is parent of person2
      if (node1 && node2) {
        node1.childrenIds.push(rel.person2Id);
        node2.parentIds.push(rel.person1Id);
        links.push({ source: rel.person1Id, target: rel.person2Id, type: 'parent' });
      }
    } else if (rel.type === 'SPOUSE' || rel.type === 'PARTNER') {
      if (node1 && node2) {
        node1.spouseIds.push(rel.person2Id);
        node2.spouseIds.push(rel.person1Id);
        links.push({ source: rel.person1Id, target: rel.person2Id, type: 'spouse' });
      }
    }
  });

  // Assign generations (BFS from root nodes)
  const visited = new Set<string>();
  const queue: Array<{ id: string; generation: number }> = [];

  // Find root nodes (no parents)
  nodeMap.forEach((node, id) => {
    if (node.parentIds.length === 0) {
      queue.push({ id, generation: 0 });
    }
  });

  while (queue.length > 0) {
    const { id, generation } = queue.shift()!;
    if (visited.has(id)) continue;
    visited.add(id);

    const node = nodeMap.get(id)!;
    node.generation = generation;

    node.childrenIds.forEach((childId) => {
      if (!visited.has(childId)) {
        queue.push({ id: childId, generation: generation + 1 });
      }
    });
  }

  // Group nodes by generation
  const generations = new Map<number, TreeNode[]>();
  nodeMap.forEach((node) => {
    if (!generations.has(node.generation)) {
      generations.set(node.generation, []);
    }
    generations.get(node.generation)!.push(node);
  });

  // Calculate positions
  generations.forEach((nodes, gen) => {
    const totalWidth = nodes.length * (NODE_WIDTH + HORIZONTAL_SPACING) - HORIZONTAL_SPACING;
    const startX = -totalWidth / 2;

    nodes.forEach((node, index) => {
      node.x = startX + index * (NODE_WIDTH + HORIZONTAL_SPACING);
      node.y = gen * (NODE_HEIGHT + VERTICAL_SPACING);
    });
  });

  return { nodes: Array.from(nodeMap.values()), links };
}
```

```typescript
// src/features/tree/components/TreeChart.tsx
import React, { useMemo, useRef, useEffect } from 'react';
import * as d3 from 'd3';
import { TreeNode, TreeLink } from '../utils/layout';

interface TreeChartProps {
  nodes: TreeNode[];
  links: TreeLink[];
  onNodeClick?: (node: TreeNode) => void;
}

const NODE_WIDTH = 180;
const NODE_HEIGHT = 80;

export function TreeChart({ nodes, links, onNodeClick }: TreeChartProps) {
  const svgRef = useRef<SVGSVGElement>(null);

  const { width, height } = useMemo(() => {
    if (nodes.length === 0) return { width: 800, height: 600 };

    const xExtent = d3.extent(nodes, (d) => d.x) as [number, number];
    const yExtent = d3.extent(nodes, (d) => d.y) as [number, number];

    return {
      width: xExtent[1] - xExtent[0] + NODE_WIDTH * 2,
      height: yExtent[1] - yExtent[0] + NODE_HEIGHT * 2,
    };
  }, [nodes]);

  useEffect(() => {
    if (!svgRef.current || nodes.length === 0) return;

    const svg = d3.select(svgRef.current);
    svg.selectAll('*').remove();

    const g = svg.append('g');

    // Draw links
    const linkGroup = g.append('g').attr('class', 'links');

    links.forEach((link) => {
      const source = nodes.find((n) => n.id === link.source);
      const target = nodes.find((n) => n.id === link.target);

      if (source && target) {
        if (link.type === 'parent') {
          // Draw parent-child link with curves
          const path = d3.path();
          path.moveTo(source.x + NODE_WIDTH / 2, source.y + NODE_HEIGHT);
          path.bezierCurveTo(
            source.x + NODE_WIDTH / 2,
            source.y + NODE_HEIGHT + (target.y - source.y - NODE_HEIGHT) / 2,
            target.x + NODE_WIDTH / 2,
            source.y + NODE_HEIGHT + (target.y - source.y - NODE_HEIGHT) / 2,
            target.x + NODE_WIDTH / 2,
            target.y
          );
          linkGroup
            .append('path')
            .attr('d', path.toString())
            .attr('fill', 'none')
            .attr('stroke', '#999')
            .attr('stroke-width', 2);
        } else {
          // Draw spouse link
          linkGroup
            .append('line')
            .attr('x1', source.x + NODE_WIDTH)
            .attr('y1', source.y + NODE_HEIGHT / 2)
            .attr('x2', target.x)
            .attr('y2', target.y + NODE_HEIGHT / 2)
            .attr('stroke', '#ff7875')
            .attr('stroke-width', 2)
            .attr('stroke-dasharray', '5,5');
        }
      }
    });

    // Draw nodes
    const nodeGroup = g.append('g').attr('class', 'nodes');

    nodes.forEach((node) => {
      const nodeG = nodeGroup
        .append('g')
        .attr('transform', `translate(${node.x}, ${node.y})`)
        .attr('class', 'node')
        .style('cursor', 'pointer')
        .on('click', () => onNodeClick?.(node));

      // Node background
      nodeG
        .append('rect')
        .attr('width', NODE_WIDTH)
        .attr('height', NODE_HEIGHT)
        .attr('rx', 8)
        .attr('fill', node.gender === 'MALE' ? '#e6f7ff' : node.gender === 'FEMALE' ? '#fff0f6' : '#f5f5f5')
        .attr('stroke', node.gender === 'MALE' ? '#1890ff' : node.gender === 'FEMALE' ? '#eb2f96' : '#d9d9d9')
        .attr('stroke-width', 2);

      // Node text
      nodeG
        .append('text')
        .attr('x', NODE_WIDTH / 2)
        .attr('y', NODE_HEIGHT / 2)
        .attr('text-anchor', 'middle')
        .attr('dominant-baseline', 'middle')
        .attr('font-size', '14px')
        .attr('font-weight', '500')
        .text(node.name.length > 18 ? node.name.slice(0, 18) + '...' : node.name);
    });

    // Center the chart
    const xExtent = d3.extent(nodes, (d) => d.x) as [number, number];
    const yExtent = d3.extent(nodes, (d) => d.y) as [number, number];
    const offsetX = -xExtent[0] + NODE_WIDTH;
    const offsetY = -yExtent[0] + NODE_HEIGHT;

    g.attr('transform', `translate(${offsetX}, ${offsetY})`);
  }, [nodes, links, onNodeClick]);

  return (
    <div className="overflow-auto border rounded-lg">
      <svg ref={svgRef} width={width + NODE_WIDTH * 2} height={height + NODE_HEIGHT * 2} />
    </div>
  );
}
```

```typescript
import { useMemo } from 'react';
import { useNavigate } from 'react-router-dom';
import { Spin, Empty } from 'antd';

import { GET_MEMBERS } from '@/features/members/graphql/queries';
import { GET_RELATIONSHIPS } from '../graphql/queries';
import { useAuth } from '@/stores/auth';

export function TreePage() {
  const { user } = useAuth();
  const navigate = useNavigate();

  const { loading: loadingMembers, data: membersData } = useQuery(GET_MEMBERS, {
    variables: { familyId: user?.familyId },
    skip: !user?.familyId,
  });

  const { loading: loadingRels, data: relsData } = useQuery(GET_RELATIONSHIPS, {
    variables: { familyId: user?.familyId },
    skip: !user?.familyId,

    if (!membersData?.members || !relsData?.relationships) {
      return { nodes: [], links: [] };
    }
    return calculateTreeLayout(membersData.members, relsData.relationships);

    return (
      <div className="flex justify-center items-center h-64">
        <Spin size="large" />
      </div>
    );

  if (nodes.length === 0) {
    return (
      <div className="p-6">
        <h1 className="text-2xl font-bold mb-6">Family Tree</h1>
        <Empty description="No family members to display" />
      </div>
    );
  }

  return (
    <div className="p-6">
      <h1 className="text-2xl font-bold mb-6">Family Tree</h1>
      <TreeChart
        nodes={nodes}
        links={links}
        onNodeClick={(node) => navigate(`/members/${node.id}`)}

    </div>
  );
}
```
