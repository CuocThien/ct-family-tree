```typescript
import { ValidationError } from '@/domain/errors';

export type TimelineEventType = 'BIRTH' | 'DEATH' | 'MARRIAGE' | 'DIVORCE' | 'GRADUATION' | 'CAREER' | 'MILESTONE' | 'CUSTOM';

interface TimelineEventProps {
  id?: string;
  familyId: string;
  eventType: TimelineEventType;
  customType?: string;
  title: string;
  description?: string;
  date: {
    year: number;
    month?: number;
    day?: number;
    isApproximate?: boolean;
  };
  location?: {
    city?: string;
    country: string;
  };
  memberIds: string[];
  mediaIds: string[];
  sourceIds: string[];
  tags: string[];
  createdBy: string;
  createdAt?: Date;
  updatedAt?: Date;
}

export class TimelineEvent {
  readonly id?: string;
  readonly familyId: string;
  readonly eventType: TimelineEventType;
  readonly customType?: string;
  readonly title: string;
  readonly description?: string;
  readonly date: {
    year: number;
    month?: number;
    day?: number;
    isApproximate?: boolean;
  };
  readonly location?: {
    city?: string;
    country: string;
  };
  readonly memberIds: string[];
  readonly mediaIds: string[];
  readonly sourceIds: string[];
  readonly tags: string[];
  readonly createdBy: string;
  readonly createdAt: Date;
  readonly updatedAt: Date;

  private constructor(props: TimelineEventProps) {
    this.id = props.id;
    this.familyId = props.familyId;
    this.eventType = props.eventType;
    this.customType = props.customType;
    this.title = props.title;
    this.description = props.description;
    this.date = props.date;
    this.location = props.location;
    this.memberIds = props.memberIds ?? [];
    this.mediaIds = props.mediaIds ?? [];
    this.sourceIds = props.sourceIds ?? [];
    this.tags = props.tags ?? [];
    this.createdBy = props.createdBy;
    this.createdAt = props.createdAt ?? new Date();
    this.updatedAt = props.updatedAt ?? new Date();
  }

  static create(props: TimelineEventProps): TimelineEvent {
    if (!props.title?.trim()) {
      throw new ValidationError('Title is required');
    }
    if (!props.date?.year) {
      throw new ValidationError('Date year is required');
    }
    if (props.eventType === 'CUSTOM' && !props.customType?.trim()) {
      throw new ValidationError('Custom type is required for CUSTOM events');
    }
    return new TimelineEvent(props);
  }

  involvesMember(memberId: string): boolean {
    return this.memberIds.includes(memberId);
  }

  getSortDate(): Date {
    return new Date(
      this.date.year,
      (this.date.month ?? 1) - 1,
      this.date.day ?? 1
    );
  }

  toJSON() {
    return {
      id: this.id,
      familyId: this.familyId,
      eventType: this.eventType,
      customType: this.customType,
      title: this.title,
      description: this.description,
      date: this.date,
      location: this.location,
      memberIds: this.memberIds,
      mediaIds: this.mediaIds,
      sourceIds: this.sourceIds,
      tags: this.tags,
      createdBy: this.createdBy,
      createdAt: this.createdAt,
      updatedAt: this.updatedAt,
    };
  }
}
```

```typescript
// src/infrastructure/database/repositories/TimelineEventRepository.ts
import { TimelineEvent, TimelineEventType } from '@/domain/entities/TimelineEvent';
import { ITimelineEventRepository, CreateTimelineEventInput, TimelineEventFilter } from '@/domain/interfaces';
import { TimelineEventModel } from '../models';
import { NotFoundError } from '@/domain/errors';

export class TimelineEventRepository implements ITimelineEventRepository {
  async findById(id: string): Promise<TimelineEvent | null> {
    const doc = await TimelineEventModel.findById(id)
      .populate('memberIds')
      .populate('mediaIds');
    return doc ? this.toEntity(doc) : null;
  }

  async findByFamilyId(familyId: string, filter?: TimelineEventFilter): Promise<TimelineEvent[]> {
    let query: any = { familyId };

    if (filter?.eventType) {
      query.eventType = filter.eventType;
    }
    if (filter?.memberId) {
      query.memberIds = filter.memberId;
    }
    if (filter?.yearFrom || filter?.yearTo) {
      query['date.year'] = {};
      if (filter?.yearFrom) query['date.year'].$gte = filter.yearFrom;
      if (filter?.yearTo) query['date.year'].$lte = filter.yearTo;
    }

    const docs = await TimelineEventModel.find(query)
      .sort({ 'date.year': 1, 'date.month': 1, 'date.day': 1 })
      .populate('memberIds')
      .populate('mediaIds');

    return docs.map(doc => this.toEntity(doc));
  }

  async findByMemberId(memberId: string): Promise<TimelineEvent[]> {
    const docs = await TimelineEventModel.find({ memberIds: memberId })
      .sort({ 'date.year': 1, 'date.month': 1, 'date.day': 1 });
    return docs.map(doc => this.toEntity(doc));
  }

  async create(input: CreateTimelineEventInput): Promise<TimelineEvent> {
    const doc = await TimelineEventModel.create(input);
    return this.toEntity(doc);
  }

  async update(id: string, updates: Partial<TimelineEvent>): Promise<TimelineEvent> {
    const doc = await TimelineEventModel.findByIdAndUpdate(
      id,
      { ...updates, updatedAt: new Date() },
      { new: true, runValidators: true }
    );
    if (!doc) throw new NotFoundError('TimelineEvent', id);
    return this.toEntity(doc);
  }

  async delete(id: string): Promise<void> {
    await TimelineEventModel.findByIdAndDelete(id);
  }

  private toEntity(doc: any): TimelineEvent {
    return TimelineEvent.create({
      id: doc._id.toString(),
      familyId: doc.familyId.toString(),
      eventType: doc.eventType as TimelineEventType,
      customType: doc.customType,
      title: doc.title,
      description: doc.description,
      date: doc.date,
      location: doc.location,
      memberIds: doc.memberIds?.map((id: any) => id.toString()) || [],
      mediaIds: doc.mediaIds?.map((id: any) => id.toString()) || [],
      sourceIds: doc.sourceIds?.map((id: any) => id.toString()) || [],
      tags: doc.tags,
      createdBy: doc.createdBy.toString(),
      createdAt: doc.createdAt,
      updatedAt: doc.updatedAt,
    });
  }
}
```

```typescript
// src/application/services/TimelineService.ts
import { ITimelineEventRepository, IUserRepository, IMemberRepository } from '@/domain/interfaces';
import { UnauthorizedError, NotFoundError } from '@/domain/errors';
import { TimelineEvent, TimelineEventType } from '@/domain/entities/TimelineEvent';

export class TimelineService {
  constructor(
    private timelineRepo: ITimelineEventRepository,
    private userRepo: IUserRepository,
    private memberRepo: IMemberRepository,
  ) {}

  async createEvent(
    userId: string,
    input: {
      familyId: string;
      eventType: TimelineEventType;
      customType?: string;
      title: string;
      description?: string;
      date: { year: number; month?: number; day?: number; isApproximate?: boolean };
      location?: { city?: string; country: string };
      memberIds?: string[];
      mediaIds?: string[];
      tags?: string[];
    }
  ): Promise<TimelineEvent> {
    const user = await this.userRepo.findById(userId);
    if (!user || !user.canEdit()) {
      throw new UnauthorizedError('Cannot create timeline events');
    }

    if (user.familyId !== input.familyId) {
      throw new UnauthorizedError('Cannot create events for other families');
    }

    // Verify all members belong to the family
    if (input.memberIds && input.memberIds.length > 0) {
      for (const memberId of input.memberIds) {
        const member = await this.memberRepo.findById(memberId);
        if (!member || member.familyId !== input.familyId) {
          throw new NotFoundError('Member', memberId);
        }
      }
    }

    return await this.timelineRepo.create({
      ...input,
      createdBy: userId,
    });
  }

  async getFamilyTimeline(
    userId: string,
    familyId: string,
    filter?: {
      eventType?: TimelineEventType;
      memberId?: string;
      yearFrom?: number;
      yearTo?: number;
    }
  ): Promise<TimelineEvent[]> {
    const user = await this.userRepo.findById(userId);
    if (!user) {
      throw new UnauthorizedError('User not found');
    }

    if (user.familyId !== familyId) {
      throw new UnauthorizedError('Cannot view timeline for other families');
    }

    return await this.timelineRepo.findByFamilyId(familyId, filter);
  }

  async deleteEvent(userId: string, eventId: string): Promise<void> {
    const user = await this.userRepo.findById(userId);
    if (!user || !user.canEdit()) {
      throw new UnauthorizedError('Cannot delete timeline events');
    }

    const event = await this.timelineRepo.findById(eventId);
    if (!event) {
      throw new NotFoundError('TimelineEvent', eventId);
    }

    if (event.familyId !== user.familyId) {
      throw new UnauthorizedError('Cannot delete events from other families');
    }

    await this.timelineRepo.delete(eventId);
  }
}
```

```typescript
    args: {
      familyId: t.arg.id({ required: true }),
      filter: t.arg({ type: TimelineFilterInput }),
    },
    resolve: async (_, { familyId, filter }, ctx) => {
      if (!ctx.user) throw new Error('Unauthorized');
      return timelineService.getFamilyTimeline(ctx.user.userId, familyId, filter ? {
        eventType: filter.eventType as any,
        memberId: filter.memberId ?? undefined,

    },

      const repo = new TimelineEventRepository();

    },
```

```typescript
import React from 'react';
import { Timeline, Card, Tag, Button, Popconfirm, Empty } from 'antd';
import {
  ClockCircleOutlined,
  HeartOutlined,
  TrophyOutlined,
  RocketOutlined,
  DeleteOutlined,
} from '@ant-design/icons';
import { useMutation } from '@apollo/client';
import { DELETE_TIMELINE_EVENT } from '../graphql/mutations';
import { useAuth } from '@/stores/auth';

interface Props {
  events: any[];
  refetch: () => void;
}

const eventIcons: Record<string, React.ReactNode> = {
  BIRTH: <HeartOutlined />,
  DEATH: <ClockCircleOutlined />,
  MARRIAGE: <HeartOutlined style={{ color: '#eb2f96' }} />,
  GRADUATION: <TrophyOutlined style={{ color: '#52c41a' }} />,
  CAREER: <RocketOutlined style={{ color: '#1890ff' }} />,
  MILESTONE: <TrophyOutlined />,
};

const eventColors: Record<string, string> = {
  BIRTH: 'green',
  DEATH: 'gray',
  MARRIAGE: 'pink',
  GRADUATION: 'blue',
  CAREER: 'purple',
  MILESTONE: 'gold',
};

export function TimelineList({ events, refetch }: Props) {
  const { canEdit } = useAuth();

  const [deleteEvent] = useMutation(DELETE_TIMELINE_EVENT, {
    onCompleted: () => refetch(),
  });

  if (events.length === 0) {
    return <Empty description="No timeline events" />;
  }

  return (
    <Timeline mode="left">
      {events.map((event: any) => {
        const dateStr = event.date.year +
          (event.date.month ? `/${event.date.month.toString().padStart(2, '0')}` : '') +
          (event.date.day ? `/${event.date.day.toString().padStart(2, '0')}` : '');

        return (
          <Timeline.Item
            key={event.id}
            dot={eventIcons[event.eventType] || <ClockCircleOutlined />}
            color={eventColors[event.eventType] || 'blue'}
            label={<span className="font-medium">{dateStr}</span>}
          >
            <Card
              size="small"
              className="mb-2"
              title={
                <div className="flex justify-between items-center">
                  <span>{event.title}</span>
                  {canEdit && (
                    <Popconfirm
                      title="Delete this event?"
                      onConfirm={() => deleteEvent({ variables: { id: event.id } })}
                    >
                      <Button size="small" icon={<DeleteOutlined />} danger />
                    </Popconfirm>
                  )}
                </div>
              }
            >
              {event.description && (
                <p className="text-gray-600 mb-2">{event.description}</p>
              )}
              {event.location && (
                <p className="text-gray-500 text-sm mb-2">
                  📍 {[event.location.city, event.location.country].filter(Boolean).join(', ')}
                </p>
              )}
              {event.tags && event.tags.length > 0 && (
                <div>
                  {event.tags.map((tag: string) => (
                    <Tag key={tag}>{tag}</Tag>
                  ))}
                </div>
              )}
            </Card>
          </Timeline.Item>
        );
      })}
    </Timeline>
  );
}
```
