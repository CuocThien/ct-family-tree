import { beforeAll, afterAll, afterEach } from 'bun:test';
import { connectToDatabase, disconnectFromDatabase } from '@/infrastructure/database/connection';
import { MongoMemoryServer } from 'mongodb-memory-server';

let mongoServer: MongoMemoryServer;

beforeAll(async () => {
  mongoServer = await MongoMemoryServer.create();
  const uri = mongoServer.getUri();
  process.env.MONGODB_URI = uri;
  await connectToDatabase();
});

afterAll(async () => {
  await disconnectFromDatabase();
  await mongoServer.stop();
});

afterEach(async () => {
  // Clean up collections between tests
  const { getMongooseInstance } = await import('@/infrastructure/database/connection');
  const mongoose = getMongooseInstance();
  const collections = mongoose.connection.collections;
  for (const key in collections) {
    await collections[key].deleteMany({});
  }
});

import { MemberModel } from '@/infrastructure/database/models/Member';

import { FamilyModel } from '@/infrastructure/database/models/Family';
import { RelationshipModel } from '@/infrastructure/database/models/Relationship';

export async function createTestFamily(name: string = 'Test Family') {
  return await FamilyModel.create({
    name,
    ownerId: null,
    settings: { isPublic: false },
  });

export async function createTestUser(overrides: Partial<any> = {}) {
  const family = await createTestFamily();
  const user = await UserModel.create({
    username: overrides.username || `testuser_${Date.now()}`,
    passwordHash: overrides.passwordHash || 'hashedpassword',
    role: overrides.role || 'OWNER',
    familyId: family._id,
    ...overrides,
  });

  // Update family owner
  await FamilyModel.findByIdAndUpdate(family._id, { ownerId: user._id });

  return { user, family };
}

export async function createTestMember(familyId: string, overrides: Partial<any> = {}) {
  return await MemberModel.create({
    familyId,
    personalInfo: {
      firstName: overrides.firstName || 'John',
      lastName: overrides.lastName || 'Doe',
      ...overrides.personalInfo,
    },
    lifeEvents: overrides.lifeEvents || [],
    ...overrides,
  });
}

    familyId,

    person1Id: overrides.person1Id,
    person2Id: overrides.person2Id,

    ...overrides,
  });

export async function cleanDatabase() {
  await Promise.all([

# Family Tree API Documentation

## Overview

Family Tree V2 provides a GraphQL API for managing family trees, members, relationships, and related data.

## Base URL

- Development: `http://localhost:3000/graphql`
- Production: `https://api.familytree.example.com/graphql`

## Authentication

All authenticated requests require a Bearer token in the Authorization header.

```
Authorization: Bearer <token>
```

See [Authentication](./authentication.md) for details.

## Schema

The API uses GraphQL with the following main types:

- **User**: User accounts and permissions
- **Family**: Family groups
- **Member**: Family members
- **Relationship**: Relationships between members
- **Media**: Uploaded files
- **Source**: Citations and references
- **TimelineEvent**: Timeline events

## Rate Limiting

- 100 requests per minute for authenticated users
- 20 requests per minute for unauthenticated users

## Error Handling

Errors follow the GraphQL spec with additional extensions:

```json
{
  "errors": [{
    "message": "Unauthorized",
    "extensions": {
      "code": "UNAUTHORIZED"
    }
  }]
}
```

## Pagination

List queries support `limit` and `skip` parameters:

```graphql
query {
  members(familyId: "xxx", limit: 20, skip: 0) {
    id
    personalInfo { firstName lastName }
  }
}
```

- [Authentication](./authentication.md)

- [Relationships](./relationships.md)
- [Timeline](./timeline.md)
