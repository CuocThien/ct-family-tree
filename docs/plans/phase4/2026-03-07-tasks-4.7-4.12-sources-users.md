```typescript
// src/domain/entities/Source.ts
import { ValidationError } from '@/domain/errors';

export type SourceType = 'DOCUMENT' | 'PHOTO' | 'VIDEO' | 'AUDIO' | 'INTERVIEW' | 'WEBSITE' | 'BOOK' | 'NEWSPAPER' | 'OFFICIAL_RECORD' | 'OTHER';
export type SourceReliability = 'HIGH' | 'MEDIUM' | 'LOW' | 'UNVERIFIED';

interface SourceProps {
  id?: string;
  familyId: string;
  type: SourceType;
  title: string;
  author?: string;
  publicationDate?: { year: number; month?: number; day?: number };
  publisher?: string;
  url?: string;
  archiveReference?: string;
  citation: string;
  reliability: SourceReliability;
  notes?: string;
  mediaIds?: string[];
  createdBy: string;
  createdAt?: Date;
  updatedAt?: Date;
}

export class Source {
  readonly id?: string;
  readonly familyId: string;
  readonly type: SourceType;
  readonly title: string;
  readonly author?: string;
  readonly publicationDate?: { year: number; month?: number; day?: number };
  readonly publisher?: string;
  readonly url?: string;
  readonly archiveReference?: string;
  readonly citation: string;
  readonly reliability: SourceReliability;
  readonly notes?: string;
  readonly mediaIds: string[];
  readonly createdBy: string;
  readonly createdAt: Date;
  readonly updatedAt: Date;

  private constructor(props: SourceProps) {
    this.id = props.id;
    this.familyId = props.familyId;
    this.type = props.type;
    this.title = props.title;
    this.author = props.author;
    this.publicationDate = props.publicationDate;
    this.publisher = props.publisher;
    this.url = props.url;
    this.archiveReference = props.archiveReference;
    this.citation = props.citation;
    this.reliability = props.reliability;
    this.notes = props.notes;
    this.mediaIds = props.mediaIds ?? [];
    this.createdBy = props.createdBy;
    this.createdAt = props.createdAt ?? new Date();
    this.updatedAt = props.updatedAt ?? new Date();
  }

  static create(props: SourceProps): Source {
    if (!props.title?.trim()) {
      throw new ValidationError('Title is required');
    }
    if (!props.citation?.trim()) {
      throw new ValidationError('Citation is required');
    }
    const validReliabilities: SourceReliability[] = ['HIGH', 'MEDIUM', 'LOW', 'UNVERIFIED'];
    if (!validReliabilities.includes(props.reliability)) {
      throw new ValidationError('Invalid reliability value');
    }
    return new Source(props);
  }

  isReliable(): boolean {
    return this.reliability === 'HIGH' || this.reliability === 'MEDIUM';
  }

  toJSON() {
    return {
      id: this.id,
      familyId: this.familyId,
      type: this.type,
      title: this.title,
      author: this.author,
      publicationDate: this.publicationDate,
      publisher: this.publisher,
      url: this.url,
      archiveReference: this.archiveReference,
      citation: this.citation,
      reliability: this.reliability,
      notes: this.notes,
      mediaIds: this.mediaIds,
      createdBy: this.createdBy,
      createdAt: this.createdAt,
      updatedAt: this.updatedAt,
    };
  }
}
```

```typescript
import { Source, SourceType, SourceReliability } from '@/domain/entities/Source';
import { ISourceRepository, CreateSourceInput, SourceFilter } from '@/domain/interfaces';
import { SourceModel } from '../models';
import { NotFoundError } from '@/domain/errors';

export class SourceRepository implements ISourceRepository {
  async findById(id: string): Promise<Source | null> {
    const doc = await SourceModel.findById(id).populate('mediaIds');
    return doc ? this.toEntity(doc) : null;
  }

  async findByFamilyId(familyId: string, filter?: SourceFilter): Promise<Source[]> {
    let query: any = { familyId };

    if (filter?.type) {
      query.type = filter.type;
    }
    if (filter?.reliability) {
      query.reliability = filter.reliability;
    }
    if (filter?.search) {
      query.$text = { $search: filter.search };
    }

    const docs = await SourceModel.find(query)
      .sort({ createdAt: -1 })
      .populate('mediaIds');

    return docs.map(doc => this.toEntity(doc));
  }

  async create(input: CreateSourceInput): Promise<Source> {
    const doc = await SourceModel.create(input);
    return this.toEntity(doc);
  }

  async update(id: string, updates: Partial<Source>): Promise<Source> {
    const doc = await SourceModel.findByIdAndUpdate(
      id,
      { ...updates, updatedAt: new Date() },
      { new: true, runValidators: true }
    );
    if (!doc) throw new NotFoundError('Source', id);
    return this.toEntity(doc);
  }

  async delete(id: string): Promise<void> {
    await SourceModel.findByIdAndDelete(id);
  }

  private toEntity(doc: any): Source {
    return Source.create({
      id: doc._id.toString(),
      familyId: doc.familyId.toString(),
      type: doc.type as SourceType,
      title: doc.title,
      author: doc.author,
      publicationDate: doc.publicationDate,
      publisher: doc.publisher,
      url: doc.url,
      archiveReference: doc.archiveReference,
      citation: doc.citation,
      reliability: doc.reliability as SourceReliability,
      notes: doc.notes,
      mediaIds: doc.mediaIds?.map((id: any) => id.toString()) || [],
      createdBy: doc.createdBy.toString(),
      createdAt: doc.createdAt,
      updatedAt: doc.updatedAt,
    });
  }
}
```

```typescript
// src/application/services/SourceService.ts
import { ISourceRepository, IUserRepository } from '@/domain/interfaces';
import { UnauthorizedError } from '@/domain/errors';
import { Source, SourceType, SourceReliability } from '@/domain/entities/Source';

export class SourceService {
  constructor(
    private sourceRepo: ISourceRepository,
    private userRepo: IUserRepository,
  ) {}

  async createSource(
    userId: string,
    input: {
      familyId: string;
      type: SourceType;
      title: string;
      author?: string;
      citation: string;
      reliability: SourceReliability;
      [key: string]: any;
    }
  ): Promise<Source> {
    const user = await this.userRepo.findById(userId);
    if (!user || !user.canEdit()) {
      throw new UnauthorizedError('Cannot create sources');
    }

    if (user.familyId !== input.familyId) {
      throw new UnauthorizedError('Cannot create sources for other families');
    }

    return await this.sourceRepo.create({
      ...input,
      createdBy: userId,
    });
  }

  async getFamilySources(
    userId: string,
    familyId: string,
    filter?: { type?: SourceType; reliability?: SourceReliability; search?: string }
  ): Promise<Source[]> {
    const user = await this.userRepo.findById(userId);
    if (!user) {
      throw new UnauthorizedError('User not found');
    }

    if (user.familyId !== familyId) {
      throw new UnauthorizedError('Cannot view sources from other families');
    }

    return await this.sourceRepo.findByFamilyId(familyId, filter);
  }

  async deleteSource(userId: string, sourceId: string): Promise<void> {
    const user = await this.userRepo.findById(userId);
    if (!user || !user.canEdit()) {
      throw new UnauthorizedError('Cannot delete sources');
    }

    const source = await this.sourceRepo.findById(sourceId);
    if (!source) {
      throw new Error('Source not found');
    }

    if (source.familyId !== user.familyId) {
      throw new UnauthorizedError('Cannot delete sources from other families');
    }

    await this.sourceRepo.delete(sourceId);
  }
}
```

```typescript
// src/application/services/UserManagementService.ts
import { IUserRepository, IFamilyRepository } from '@/domain/interfaces';
import { UnauthorizedError, NotFoundError, ValidationError, ConflictError } from '@/domain/errors';
import { User, UserRole } from '@/domain/entities/User';
import { hashPassword } from '@/utils/hash';

export class UserManagementService {
  constructor(
    private userRepo: IUserRepository,
    private familyRepo: IFamilyRepository,
  ) {}

  async createUser(
    ownerId: string,
    input: {
      username: string;
      password: string;
      role: UserRole;
      familyId: string;
      memberProfileId?: string;
    }
  ): Promise<User> {
    // Verify owner
    const owner = await this.userRepo.findById(ownerId);
    if (!owner || !owner.isOwner()) {
      throw new UnauthorizedError('Only family owners can create users');
    }

    // Verify same family
    if (owner.familyId !== input.familyId) {
      throw new UnauthorizedError('Cannot create users for other families');
    }

    // Check username uniqueness
    const existing = await this.userRepo.findByUsername(input.username);
    if (existing) {
      throw new ConflictError('Username already exists');
    }

    // Validate password
    if (input.password.length < 8) {
      throw new ValidationError('Password must be at least 8 characters');
    }

    const passwordHash = await hashPassword(input.password);

    return await this.userRepo.create({
      username: input.username,
      passwordHash,
      role: input.role,
      familyId: input.familyId,
      memberProfileId: input.memberProfileId,
    });
  }

  async updateUserRole(
    ownerId: string,
    userId: string,
    newRole: UserRole
  ): Promise<User> {
    const owner = await this.userRepo.findById(ownerId);
    if (!owner || !owner.isOwner()) {
      throw new UnauthorizedError('Only family owners can update user roles');
    }

    const user = await this.userRepo.findById(userId);
    if (!user) {
      throw new NotFoundError('User', userId);
    }

    if (user.familyId !== owner.familyId) {
      throw new UnauthorizedError('Cannot update users from other families');
    }

    // Cannot change own role
    if (userId === ownerId) {
      throw new ValidationError('Cannot change your own role');
    }

    return await this.userRepo.update(userId, { role: newRole } as any);
  }

  async deleteUser(ownerId: string, userId: string): Promise<void> {
    const owner = await this.userRepo.findById(ownerId);
    if (!owner || !owner.isOwner()) {
      throw new UnauthorizedError('Only family owners can delete users');
    }

    const user = await this.userRepo.findById(userId);
    if (!user) {
      throw new NotFoundError('User', userId);
    }

    if (user.familyId !== owner.familyId) {
      throw new UnauthorizedError('Cannot delete users from other families');
    }

    // Cannot delete self
    if (userId === ownerId) {
      throw new ValidationError('Cannot delete your own account');
    }

    await this.userRepo.delete(userId);
  }

  async resetUserPassword(
    ownerId: string,
    userId: string,
    newPassword: string
  ): Promise<void> {
    const owner = await this.userRepo.findById(ownerId);
    if (!owner || !owner.isOwner()) {
      throw new UnauthorizedError('Only family owners can reset passwords');
    }

    const user = await this.userRepo.findById(userId);
    if (!user) {
      throw new NotFoundError('User', userId);
    }

    if (user.familyId !== owner.familyId) {
      throw new UnauthorizedError('Cannot reset passwords for users from other families');
    }

    if (newPassword.length < 8) {
      throw new ValidationError('Password must be at least 8 characters');
    }

    const passwordHash = await hashPassword(newPassword);
    await this.userRepo.update(userId, { passwordHash } as any);
  }

  async getFamilyUsers(ownerId: string, familyId: string): Promise<User[]> {
    const owner = await this.userRepo.findById(ownerId);
    if (!owner) {
      throw new UnauthorizedError('User not found');
    }

    if (owner.familyId !== familyId) {
      throw new UnauthorizedError('Cannot view users from other families');
    }

    return await this.userRepo.findByFamilyId(familyId);
  }
}
```

```typescript
    role: t.string({ required: true }),
    familyId: t.id({ required: true }),
    memberProfileId: t.id(),
  }),
});

  fields: (t) => ({
    userId: t.id({ required: true }),
    newRole: t.string({ required: true }),

      return userManagementService.getFamilyUsers(ctx.user.userId, familyId);
    },
  }),
}));

  createUser: t.field({
    type: 'UserManagement',
    args: {
      input: t.arg({ type: CreateUserInput, required: true }),
    },
    resolve: async (_, { input }, ctx) => {
      if (!ctx.user) throw new Error('Unauthorized');
      return userManagementService.createUser(ctx.user.userId, {
        username: input.username,
        password: input.password,
        role: input.role as any,
        familyId: input.familyId,
        memberProfileId: input.memberProfileId ?? undefined,
      });
    },
  }),

  updateUserRole: t.field({
    type: 'UserManagement',
    args: {
      input: t.arg({ type: UpdateUserRoleInput, required: true }),

      if (!ctx.user) throw new Error('Unauthorized');

      input: t.arg({ type: ResetPasswordInput, required: true }),

        ctx.user.userId,
```

```typescript
// src/features/users/pages/UsersPage.tsx
import { useState } from 'react';
import { Table, Button, Tag, Modal, Select, message, Typography, Card } from 'antd';
import { PlusOutlined, KeyOutlined, DeleteOutlined } from '@ant-design/icons';
import { useQuery, useMutation } from '@apollo/client';
import { UserForm } from '../components/UserForm';
import { GET_FAMILY_USERS, UPDATE_USER_ROLE, DELETE_USER, RESET_PASSWORD } from '../graphql';
import { useAuth } from '@/stores/auth';

const { Title } = Typography;
const { Option } = Select;

export function UsersPage() {
  const { user } = useAuth();
  const [formVisible, setFormVisible] = useState(false);

  const { loading, data, refetch } = useQuery(GET_FAMILY_USERS, {
    variables: { familyId: user?.familyId },
    skip: !user?.familyId,
  });

  const [updateRole] = useMutation(UPDATE_USER_ROLE, {
    onCompleted: () => {
      message.success('Role updated');
      refetch();
    },
  });

  const [deleteUser] = useMutation(DELETE_USER, {
    onCompleted: () => {
      message.success('User deleted');
      refetch();
    },
  });

  const [resetPassword] = useMutation(RESET_PASSWORD, {
    onCompleted: () => message.success('Password reset'),
  });

  const users = data?.familyUsers || [];

  const isOwner = user?.role === 'OWNER';

  const handleRoleChange = (userId: string, newRole: string) => {
    updateRole({
      variables: {
        input: { userId, newRole },
      },
    });
  };

  const handleDelete = (userId: string) => {
    Modal.confirm({
      title: 'Delete User',
      content: 'Are you sure you want to delete this user?',
      onOk: () => deleteUser({ variables: { userId } }),
    });
  };

  const handleResetPassword = (userId: string) => {
    Modal.confirm({
      title: 'Reset Password',
      content: (
        <div>
          <p>Enter new password:</p>
          <input
            type="password"
            id="new-password-input"
            className="border p-2 w-full"
            placeholder="New password"
          />
        </div>
      ),
      onOk: () => {
        const input = document.getElementById('new-password-input') as HTMLInputElement;
        const newPassword = input?.value;
        if (newPassword && newPassword.length >= 8) {
          resetPassword({
            variables: {
              input: { userId, newPassword },
            },
          });
        } else {
          message.error('Password must be at least 8 characters');
        }
      },
    });
  };

  const roleColors: Record<string, string> = {
    OWNER: 'gold',
    EDITOR: 'blue',
    VIEWER: 'default',
  };

  const columns = [
    {
      title: 'Username',
      dataIndex: 'username',
      render: (name: string) => <strong>{name}</strong>,
    },
    {
      title: 'Role',
      dataIndex: 'role',
      width: 150,
      render: (role: string, record: any) =>
        isOwner && record.id !== user?.id ? (
          <Select
            value={role}
            onChange={(v) => handleRoleChange(record.id, v)}
            style={{ width: 120 }}
          >
            <Option value="OWNER">OWNER</Option>
            <Option value="EDITOR">EDITOR</Option>
            <Option value="VIEWER">VIEWER</Option>
          </Select>
        ) : (
          <Tag color={roleColors[role]}>{role}</Tag>
        ),
    },
    {
      title: 'Created',
      dataIndex: 'createdAt',
      render: (date: string) => new Date(date).toLocaleDateString(),
    },
    {
      title: 'Actions',
      key: 'actions',
      width: 150,
      render: (_: any, record: any) =>
        isOwner && record.id !== user?.id && (
          <div className="flex gap-2">
            <Button
              icon={<KeyOutlined />}
              size="small"
              onClick={() => handleResetPassword(record.id)}
            >
              Reset
            </Button>
            <Button
              icon={<DeleteOutlined />}
              danger
              size="small"
              onClick={() => handleDelete(record.id)}
            />
          </div>
        ),
    },
  ];

  return (
    <div className="p-6">
      <div className="flex justify-between items-center mb-6">
        <Title level={2}>User Management</Title>
        {isOwner && (
          <Button type="primary" icon={<PlusOutlined />} onClick={() => setFormVisible(true)}>
            Add User
          </Button>
        )}
      </div>

      <Card>
        <Table
          columns={columns}
          dataSource={users}
          rowKey="id"
          loading={loading}
          pagination={false}
        />
      </Card>

      <Modal
        title="Add User"
        open={formVisible}
        onCancel={() => setFormVisible(false)}
        footer={null}
      >
        <UserForm
          familyId={user?.familyId || ''}
          onSuccess={() => {
            setFormVisible(false);
            refetch();
          }}
        />
      </Modal>
    </div>
  );
}
```
