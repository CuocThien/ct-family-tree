  DomainError,
  NotFoundError,
  UnauthorizedError,
  ValidationError,
  ConflictError,
  ForbiddenError,

describe('Domain Errors', () => {
  test('DomainError should have correct name and message', () => {

  });

  test('UnauthorizedError should have default message', () => {
    const error = new UnauthorizedError();

    expect(error.message).toBe('Unauthorized access');

## Step 4: Create DomainError base class

Run:
```bash
cat > /Users/CuocThien/Desktop/NCT/WEB/FamilyTreeV2/backend/src/domain/errors/DomainError.ts << 'EOF'
export class DomainError extends Error {
  constructor(message: string) {
    super(message);
    this.name = 'DomainError';
    // Maintains proper stack trace for where error was thrown (only in V8)
    if (Error.captureStackTrace) {
      Error.captureStackTrace(this, DomainError);
    }
  }
}
EOF
```
