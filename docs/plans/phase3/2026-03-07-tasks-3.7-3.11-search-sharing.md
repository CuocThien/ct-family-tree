// Update MemberRepository with advanced search
async advancedSearch(
  familyId: string,
  options: {
    query?: string;
    gender?: 'MALE' | 'FEMALE' | 'OTHER';
    isLiving?: boolean;
    birthYearFrom?: number;
    birthYearTo?: number;
    limit?: number;
    skip?: number;
  }
): Promise<{ members: Member[]; totalCount: number }> {
  const limit = options.limit ?? 20;
  const skip = options.skip ?? 0;

  // Build query
  const query: any = { familyId };

  if (options.gender) {
    query['personalInfo.gender'] = options.gender;
  }

  if (options.query) {
    query.$text = { $search: options.query };
  }

  // Get all matching members
  let docs = await MemberModel.find(query)
    .sort(options.query ? { score: { $meta: 'textScore' } } : { 'personalInfo.lastName': 1 })
    .skip(skip)
    .limit(limit);

  // Filter by living status and birth year range (in-memory)
  if (options.isLiving !== undefined || options.birthYearFrom || options.birthYearTo) {
    docs = docs.filter(doc => {
      // Living status filter
      if (options.isLiving !== undefined) {
        const hasDeath = doc.lifeEvents.some(e => e.type === 'DEATH');
        if (options.isLiving ? hasDeath : !hasDeath) return false;
      }

      // Birth year range filter
      const birthEvent = doc.lifeEvents.find(e => e.type === 'BIRTH');
      if (birthEvent?.date?.year) {
        if (options.birthYearFrom && birthEvent.date.year < options.birthYearFrom) return false;
        if (options.birthYearTo && birthEvent.date.year > options.birthYearTo) return false;
      }

      return true;
    });
  }

  const totalCount = await MemberModel.countDocuments(query);

  return {
    members: docs.map(doc => this.toEntity(doc)),
    totalCount,
  };
}

import { ValidationError } from '@/domain/errors';
import { randomBytes } from 'crypto';

export type SharePermission = 'VIEW' | 'EDIT';

interface ShareLinkProps {
  id?: string;
  familyId: string;
  token: string;
  permission: SharePermission;
  passwordHash?: string;
  expiresAt?: Date;
  maxUses?: number;
  currentUses: number;
  createdBy: string;
  createdAt?: Date;
  updatedAt?: Date;
}

export class ShareLink {
  readonly id?: string;
  readonly familyId: string;
  readonly token: string;
  readonly permission: SharePermission;
  readonly passwordHash?: string;
  readonly expiresAt?: Date;
  readonly maxUses?: number;
  readonly currentUses: number;
  readonly createdBy: string;
  readonly createdAt: Date;
  readonly updatedAt: Date;

  private constructor(props: ShareLinkProps) {
    this.id = props.id;
    this.familyId = props.familyId;
    this.token = props.token;
    this.permission = props.permission;
    this.passwordHash = props.passwordHash;
    this.expiresAt = props.expiresAt;
    this.maxUses = props.maxUses;
    this.currentUses = props.currentUses;
    this.createdBy = props.createdBy;
    this.createdAt = props.createdAt ?? new Date();
    this.updatedAt = props.updatedAt ?? new Date();
  }

  static create(props: Omit<ShareLinkProps, 'token' | 'currentUses'>): ShareLink {
    const validPermissions: SharePermission[] = ['VIEW', 'EDIT'];
    if (!validPermissions.includes(props.permission)) {
      throw new ValidationError('Invalid permission');
    }
    const token = randomBytes(16).toString('hex');
    return new ShareLink({
      ...props,
      token,
      currentUses: 0,
    });
  }

  isValid(): boolean {
    // Check expiration
    if (this.expiresAt && new Date() > this.expiresAt) {
      return false;
    }
    // Check usage limit
    if (this.maxUses !== undefined && this.currentUses >= this.maxUses) {
      return false;
    }
    return true;
  }

  hasPassword(): boolean {
    return !!this.passwordHash;
  }

  canEdit(): boolean {
    return this.permission === 'EDIT';
  }

  incrementUse(): ShareLink {
    return new ShareLink({
      ...this,
      currentUses: this.currentUses + 1,
      updatedAt: new Date(),
    });
  }

  toJSON() {
    return {
      id: this.id,
      familyId: this.familyId,
      token: this.token,
      permission: this.permission,
      hasPassword: this.hasPassword(),
      expiresAt: this.expiresAt,
      maxUses: this.maxUses,
      currentUses: this.currentUses,
      createdBy: this.createdBy,
      createdAt: this.createdAt,
      updatedAt: this.updatedAt,
    };
  }
}

// src/application/services/ShareService.ts
import { IShareLinkRepository, IUserRepository, IFamilyRepository } from '@/domain/interfaces';
import { UnauthorizedError, NotFoundError, ValidationError } from '@/domain/errors';
import { ShareLink, SharePermission } from '@/domain/entities/ShareLink';
import { hashPassword, verifyPassword } from '@/utils/hash';

export class ShareService {
  constructor(
    private shareLinkRepo: IShareLinkRepository,
    private userRepo: IUserRepository,
    private familyRepo: IFamilyRepository,
  ) {}

  async createShareLink(
    userId: string,
    input: {
      familyId: string;
      permission: SharePermission;
      password?: string;
      expiresInDays?: number;
      maxUses?: number;
    }
  ): Promise<ShareLink> {
    const user = await this.userRepo.findById(userId);
    if (!user || !user.isOwner()) {
      throw new UnauthorizedError('Only family owners can create share links');
    }

    if (user.familyId !== input.familyId) {
      throw new UnauthorizedError('Cannot create share links for other families');
    }

    let expiresAt: Date | undefined;
    if (input.expiresInDays) {
      expiresAt = new Date();
      expiresAt.setDate(expiresAt.getDate() + input.expiresInDays);
    }

    let passwordHash: string | undefined;
    if (input.password) {
      passwordHash = await hashPassword(input.password);
    }

    return await this.shareLinkRepo.create({
      familyId: input.familyId,
      permission: input.permission,
      passwordHash,
      expiresAt,
      maxUses: input.maxUses,
      createdBy: userId,
    });
  }

  async accessSharedFamily(
    token: string,
    password?: string
  ): Promise<{ familyId: string; permission: SharePermission }> {
    const shareLink = await this.shareLinkRepo.findByToken(token);
    if (!shareLink) {
      throw new NotFoundError('ShareLink', token);
    }

    if (!shareLink.isValid()) {
      throw new ValidationError('Share link has expired or reached usage limit');
    }

    if (shareLink.hasPassword()) {
      if (!password) {
        throw new ValidationError('Password required');
      }
      const valid = await verifyPassword(password, shareLink.passwordHash!);
      if (!valid) {
        throw new UnauthorizedError('Invalid password');
      }
    }

    // Increment usage
    await this.shareLinkRepo.update(shareLink.id!, {
      currentUses: shareLink.currentUses + 1,
    });

    return {
      familyId: shareLink.familyId,
      permission: shareLink.permission,
    };
  }

  async getFamilyShareLinks(userId: string, familyId: string): Promise<ShareLink[]> {
    const user = await this.userRepo.findById(userId);
    if (!user) {
      throw new UnauthorizedError('User not found');
    }

    if (user.familyId !== familyId) {
      throw new UnauthorizedError('Cannot view share links for other families');
    }

    return await this.shareLinkRepo.findByFamilyId(familyId);
  }

  async deleteShareLink(userId: string, shareLinkId: string): Promise<void> {
    const user = await this.userRepo.findById(userId);
    if (!user || !user.isOwner()) {
      throw new UnauthorizedError('Only family owners can delete share links');
    }

    const shareLink = await this.shareLinkRepo.findById(shareLinkId);
    if (!shareLink) {
      throw new NotFoundError('ShareLink', shareLinkId);
    }

    if (shareLink.familyId !== user.familyId) {
      throw new UnauthorizedError('Cannot delete share links for other families');
    }

    await this.shareLinkRepo.delete(shareLinkId);
  }
}

      dataIndex: 'token',
      key: 'token',
      render: (token: string) => (
        <Text copyable={{ text: `${window.location.origin}/shared/${token}` }}>
          ...{token.slice(-8)}
        </Text>
      ),
    },
    {
      title: 'Protected',
      dataIndex: 'hasPassword',
      key: 'hasPassword',
      render: (has: boolean) => (
        <Tag color={has ? 'green' : 'default'}>{has ? 'Yes' : 'No'}</Tag>
      ),
    },
    {
      title: 'Uses',
      key: 'uses',
      render: (_: any, record: any) => (
        <span>
          {record.currentUses}
          {record.maxUses ? ` / ${record.maxUses}` : ''}
        </span>
      ),
    },
    {
      title: 'Status',
      dataIndex: 'isValid',
      key: 'isValid',
      render: (valid: boolean) => (
        <Tag color={valid ? 'green' : 'red'}>{valid ? 'Active' : 'Expired'}</Tag>
      ),
    },
    {
      title: 'Actions',
      key: 'actions',
      render: (_: any, record: any) => (
        <Space>
          <Button
            icon={<CopyOutlined />}
            onClick={() => copyLink(record.token)}
          />
          <Popconfirm
            title="Delete this share link?"
            onConfirm={() => deleteShareLink({
              variables: { id: record.id },
              refetchQueries: [{ query: GET_SHARE_LINKS, variables: { familyId } }],
            })}
          >
            <Button icon={<DeleteOutlined />} danger />
          </Popconfirm>
        </Space>
      ),
    },
  ];

  return (
    <Table
      columns={columns}
      dataSource={shareLinks}
      rowKey="id"
      pagination={false}
    />
  );
}
