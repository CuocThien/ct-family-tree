// src/application/services/RelationshipCalculator.ts
import { Member } from '@/domain/entities';
import { Relationship } from '@/domain/entities/Relationship';

export type CalculatedRelationship = {
  fromMemberId: string;
  toMemberId: string;
  relationship: string;
  degree: number;
  path: string[];
};

export class RelationshipCalculator {
  private memberMap: Map<string, Member>;
  private relationships: Relationship[];

  constructor(members: Member[], relationships: Relationship[]) {
    this.memberMap = new Map(members.map(m => [m.id!, m]));
    this.relationships = relationships;
  }

  /**
   * Build adjacency list for graph traversal
   */
  private buildGraph(): Map<string, Array<{ id: string; type: string }>> {
    const graph = new Map<string, Array<{ id: string; type: string }>>();

    // Initialize all members
    this.memberMap.forEach((_, id) => {
      graph.set(id, []);
    });

    // Add relationships as edges
    this.relationships.forEach(rel => {
      const edges1 = graph.get(rel.person1Id) || [];
      const edges2 = graph.get(rel.person2Id) || [];

      edges1.push({ id: rel.person2Id, type: rel.type });
      edges2.push({ id: rel.person1Id, type: rel.type });

      graph.set(rel.person1Id, edges1);
      graph.set(rel.person2Id, edges2);
    });

    return graph;
  }

  /**
   * Find shortest path between two members using BFS
   */
  findShortestPath(fromId: string, toId: string): string[] | null {
    if (fromId === toId) return [fromId];

    const graph = this.buildGraph();
    const visited = new Set<string>();
    const queue: Array<{ id: string; path: string[] }> = [{ id: fromId, path: [fromId] }];

    while (queue.length > 0) {
      const { id, path } = queue.shift()!;

      if (visited.has(id)) continue;
      visited.add(id);

      const neighbors = graph.get(id) || [];
      for (const neighbor of neighbors) {
        if (neighbor.id === toId) {
          return [...path, neighbor.id];
        }
        if (!visited.has(neighbor.id)) {
          queue.push({ id: neighbor.id, path: [...path, neighbor.id] });
        }
      }
    }

    return null;
  }

  /**
   * Calculate relationship between two members
   */
  calculateRelationship(fromId: string, toId: string): CalculatedRelationship | null {
    const path = this.findShortestPath(fromId, toId);
    if (!path || path.length < 2) return null;

    const relationship = this.determineRelationshipType(path);
    const degree = path.length - 1;

    return {
      fromMemberId: fromId,
      toMemberId: toId,
      relationship,
      degree,
      path,
    };
  }

  /**
   * Determine relationship type from path
   */
  private determineRelationshipType(path: string[]): string {
    if (path.length === 2) {
      // Direct relationship
      const rel = this.findRelationship(path[0], path[1]);
      if (rel) {
        if (rel.type === 'SPOUSE' || rel.type === 'PARTNER') {
          return rel.type === 'SPOUSE' ? 'Spouse' : 'Partner';
        }
        if (rel.type === 'PARENT') {
          // Determine if person1 is parent of person2 or vice versa
          const member1 = this.memberMap.get(path[0]);
          const member2 = this.memberMap.get(path[1]);
          if (member1 && member2) {
            const birth1 = member1.lifeEvents.find(e => e.type === 'BIRTH');
            const birth2 = member2.lifeEvents.find(e => e.type === 'BIRTH');
            if (birth1?.date?.year && birth2?.date?.year) {
              if (birth1.date.year < birth2.date.year) {
                return 'Parent';
              }
              return 'Child';
            }
          }
        }
      }
    }

    if (path.length === 3) {
      // Grandparent/Grandchild or Sibling or Uncle/Aunt/Nephew/Niece
      const rel1 = this.findRelationship(path[0], path[1]);
      const rel2 = this.findRelationship(path[1], path[2]);

      if (rel1?.type === 'PARENT' && rel2?.type === 'PARENT') {
        return this.isOlderGeneration(path[0], path[2]) ? 'Grandparent' : 'Grandchild';
      }
      if (rel1?.type === 'SPOUSE' && rel2?.type === 'PARENT') {
        return this.isOlderGeneration(path[0], path[2]) ? 'Grandparent' : 'Grandchild';
      }
    }

    // Generic relationship based on degree
    const degree = path.length - 1;
    if (degree === 2) return '2nd Degree Relative';
    if (degree === 3) return '3rd Degree Relative';
    if (degree === 4) return '4th Degree Relative';
    return `${degree}th Degree Relative`;
  }

  private findRelationship(person1Id: string, person2Id: string): Relationship | undefined {
    return this.relationships.find(
      r => (r.person1Id === person1Id && r.person2Id === person2Id) ||
           (r.person1Id === person2Id && r.person2Id === person1Id)
    );
  }

  private isOlderGeneration(person1Id: string, person2Id: string): boolean {
    const member1 = this.memberMap.get(person1Id);
    const member2 = this.memberMap.get(person2Id);

    if (!member1 || !member2) return false;

    const birth1 = member1.lifeEvents.find(e => e.type === 'BIRTH');
    const birth2 = member2.lifeEvents.find(e => e.type === 'BIRTH');

    if (!birth1?.date?.year || !birth2?.date?.year) return false;

    return birth1.date.year < birth2.date.year;
  }

  /**
   * Get all relationships for a member
   */
  getAllRelationships(memberId: string): CalculatedRelationship[] {
    const results: CalculatedRelationship[] = [];

    this.memberMap.forEach((_, otherId) => {
      if (otherId !== memberId) {
        const rel = this.calculateRelationship(memberId, otherId);
        if (rel) {
          results.push(rel);
        }
      }
    });

    return results;
  }
}

      familyId: t.arg.id({ required: true }),

// src/application/services/RelationshipService.ts
import { IRelationshipRepository, IMemberRepository, IUserRepository } from '@/domain/interfaces';
import { UnauthorizedError, NotFoundError, ValidationError } from '@/domain/errors';
import { Relationship, RelationshipType } from '@/domain/entities/Relationship';

export interface CreateRelationshipInput {
  familyId: string;
  type: RelationshipType;
  person1Id: string;
  person2Id: string;
  startDate?: Date;
  endDate?: Date;
  isCurrent: boolean;
  metadata?: Record<string, any>;
}

export class RelationshipService {
  constructor(
    private relRepo: IRelationshipRepository,
    private memberRepo: IMemberRepository,
    private userRepo?: IUserRepository,
  ) {}

  async createRelationship(userId: string, input: CreateRelationshipInput): Promise<Relationship> {
    // Verify both members exist and belong to the same family
    const member1 = await this.memberRepo.findById(input.person1Id);
    const member2 = await this.memberRepo.findById(input.person2Id);

    if (!member1 || !member2) {
      throw new NotFoundError('Member', !member1 ? input.person1Id : input.person2Id);
    }

    if (member1.familyId !== member2.familyId) {
      throw new ValidationError('Members must belong to the same family');
    }

    if (member1.familyId !== input.familyId) {
      throw new ValidationError('Members do not belong to the specified family');
    }

    // Check for existing relationship of same type
    const existingRels = await this.relRepo.findByMembers(input.person1Id, input.person2Id);
    const sameTypeRel = existingRels.find(r => r.type === input.type && r.isCurrent);
    if (sameTypeRel) {
      throw new ValidationError(`A current ${input.type} relationship already exists between these members`);
    }

    // Validate relationship type constraints
    if (input.type === 'PARENT') {
      // Verify the parent is older (if birth dates are known)
      const birth1 = member1.lifeEvents.find(e => e.type === 'BIRTH');
      const birth2 = member2.lifeEvents.find(e => e.type === 'BIRTH');

      if (birth1?.date?.year && birth2?.date?.year) {
        if (birth1.date.year >= birth2.date.year) {
          throw new ValidationError('Parent must be older than child');
        }
      }
    }

    return await this.relRepo.create(input);
  }

  async deleteRelationship(userId: string, relationshipId: string): Promise<void> {
    const relationship = await this.relRepo.findById(relationshipId);
    if (!relationship) {
      throw new NotFoundError('Relationship', relationshipId);
    }

    await this.relRepo.delete(relationshipId);
  }

  async getRelationships(familyId: string): Promise<Relationship[]> {
    return await this.relRepo.findByFamilyId(familyId);
  }

  async getMemberRelationships(memberId: string): Promise<Relationship[]> {
    return await this.relRepo.findByMemberId(memberId);
  }
}

// src/infrastructure/database/repositories/RelationshipRepository.ts
import { Relationship, RelationshipType } from '@/domain/entities/Relationship';
import { IRelationshipRepository, CreateRelationshipInput } from '@/domain/interfaces';
import { RelationshipModel } from '../models';
import { NotFoundError } from '@/domain/errors';

export class RelationshipRepository implements IRelationshipRepository {
  async findById(id: string): Promise<Relationship | null> {
    const doc = await RelationshipModel.findById(id);
    return doc ? this.toEntity(doc) : null;
  }

  async findByFamilyId(familyId: string): Promise<Relationship[]> {
    const docs = await RelationshipModel.find({ familyId });
    return docs.map(doc => this.toEntity(doc));
  }

  async findByMemberId(memberId: string): Promise<Relationship[]> {
    const docs = await RelationshipModel.find({
      $or: [{ person1Id: memberId }, { person2Id: memberId }],
    });
    return docs.map(doc => this.toEntity(doc));
  }

  async findByMembers(person1Id: string, person2Id: string): Promise<Relationship[]> {
    const docs = await RelationshipModel.find({
      $or: [
        { person1Id, person2Id },
        { person1Id: person2Id, person2Id: person1Id },
      ],
    });
    return docs.map(doc => this.toEntity(doc));
  }

  async create(input: CreateRelationshipInput): Promise<Relationship> {
    const doc = await RelationshipModel.create(input);
    return this.toEntity(doc);
  }

  async update(id: string, updates: Partial<Relationship>): Promise<Relationship> {
    const doc = await RelationshipModel.findByIdAndUpdate(
      id,
      { ...updates, updatedAt: new Date() },
      { new: true, runValidators: true }
    );
    if (!doc) throw new NotFoundError('Relationship', id);
    return this.toEntity(doc);
  }

  async delete(id: string): Promise<void> {
    await RelationshipModel.findByIdAndDelete(id);
  }

  private toEntity(doc: any): Relationship {
    return Relationship.create({
      id: doc._id.toString(),
      familyId: doc.familyId.toString(),
      type: doc.type as RelationshipType,
      person1Id: doc.person1Id.toString(),
      person2Id: doc.person2Id.toString(),
      startDate: doc.startDate,
      endDate: doc.endDate,
      isCurrent: doc.isCurrent,
      metadata: doc.metadata,
      createdAt: doc.createdAt,
      updatedAt: doc.updatedAt,
    });
  }
}

import { ValidationError } from '@/domain/errors';

export type RelationshipType = 'PARENT' | 'SPOUSE' | 'PARTNER' | 'ADOPTION';

interface RelationshipProps {
  id?: string;
  familyId: string;
  type: RelationshipType;
  person1Id: string;
  person2Id: string;
  startDate?: Date;
  endDate?: Date;
  isCurrent: boolean;
  metadata?: Record<string, any>;
  createdAt?: Date;
  updatedAt?: Date;
}

export class Relationship {
  readonly id?: string;
  readonly familyId: string;
  readonly type: RelationshipType;
  readonly person1Id: string;
  readonly person2Id: string;
  readonly startDate?: Date;
  readonly endDate?: Date;
  readonly isCurrent: boolean;
  readonly metadata?: Record<string, any>;
  readonly createdAt: Date;
  readonly updatedAt: Date;

  private constructor(props: RelationshipProps) {
    this.id = props.id;
    this.familyId = props.familyId;
    this.type = props.type;
    this.person1Id = props.person1Id;
    this.person2Id = props.person2Id;
    this.startDate = props.startDate;
    this.endDate = props.endDate;
    this.isCurrent = props.isCurrent;
    this.metadata = props.metadata;
    this.createdAt = props.createdAt ?? new Date();
    this.updatedAt = props.updatedAt ?? new Date();
  }

  static create(props: RelationshipProps): Relationship {
    if (props.person1Id === props.person2Id) {
      throw new ValidationError('Cannot create self-relationship');
    }
    const validTypes: RelationshipType[] = ['PARENT', 'SPOUSE', 'PARTNER', 'ADOPTION'];
    if (!validTypes.includes(props.type)) {
      throw new ValidationError('Invalid relationship type');
    }
    return new Relationship(props);
  }

  involves(memberId: string): boolean {
    return this.person1Id === memberId || this.person2Id === memberId;
  }

  getOtherMemberId(memberId: string): string {
    if (this.person1Id === memberId) return this.person2Id;
    if (this.person2Id === memberId) return this.person1Id;
    throw new Error('Member not involved in this relationship');
  }

  toJSON() {
    return {
      id: this.id,
      familyId: this.familyId,
      type: this.type,
      person1Id: this.person1Id,
      person2Id: this.person2Id,
      startDate: this.startDate,
      endDate: this.endDate,
      isCurrent: this.isCurrent,
      metadata: this.metadata,
      createdAt: this.createdAt,
      updatedAt: this.updatedAt,
    };
  }
}

  endDate: { type: Date },
  isCurrent: { type: Boolean, default: true },
  metadata: { type: Schema.Types.Mixed },
}, { timestamps: true });

// Compound index for finding relationships between members
RelationshipSchema.index({ person1Id: 1, person2Id: 1 });
RelationshipSchema.index({ familyId: 1, type: 1 });

export const RelationshipModel = mongoose.model<RelationshipDocument>('Relationship', RelationshipSchema);

// src/features/relationships/components/RelationshipForm.tsx
import { useState } from 'react';
import { Form, Select, Button, DatePicker, Switch, message, Modal } from 'antd';
import { useMutation, useQuery } from '@apollo/client';
import { CREATE_RELATIONSHIP } from '../graphql/mutations';
import { GET_MEMBERS } from '@/features/members/graphql/queries';
import { useAuth } from '@/stores/auth';

const { Option } = Select;

interface RelationshipFormProps {
  visible: boolean;
  onClose: () => void;
  initialMemberId?: string;
}

export function RelationshipForm({ visible, onClose, initialMemberId }: RelationshipFormProps) {
  const [form] = Form.useForm();
  const { user } = useAuth();

  const { data: membersData } = useQuery(GET_MEMBERS, {
    variables: { familyId: user?.familyId },
    skip: !user?.familyId,
  });

  const [createRelationship, { loading }] = useMutation(CREATE_RELATIONSHIP, {
    onCompleted: () => {
      message.success('Relationship created');
      form.resetFields();
      onClose();
    },
    onError: (error) => {
      message.error(error.message);
    },
    refetchQueries: ['GetRelationships'],
  });

  const handleSubmit = async (values: any) => {
    await createRelationship({
      variables: {
        input: {
          type: values.type,
          person1Id: values.person1Id,
          person2Id: values.person2Id,
          isCurrent: values.isCurrent ?? true,
        },
      },
    });
  };

  const members = membersData?.members || [];

  return (
    <Modal
      title="Add Relationship"
      open={visible}
      onCancel={onClose}
      footer={null}
    >
      <Form
        form={form}
        layout="vertical"
        onFinish={handleSubmit}
        initialValues={{
          person1Id: initialMemberId,
          isCurrent: true,
        }}
      >
        <Form.Item
          name="type"
          label="Relationship Type"
          rules={[{ required: true, message: 'Please select type' }]}
        >
          <Select placeholder="Select relationship type">
            <Option value="PARENT">Parent - Child</Option>
            <Option value="SPOUSE">Spouse</Option>
            <Option value="PARTNER">Partner</Option>
            <Option value="ADOPTION">Adoption</Option>
          </Select>
        </Form.Item>

        <Form.Item
          name="person1Id"
          label="First Person"
          rules={[{ required: true, message: 'Please select first person' }]}
        >
          <Select placeholder="Select first person" showSearch optionFilterProp="children">
            {members.map((m: any) => (
              <Option key={m.id} value={m.id}>
                {m.personalInfo.firstName} {m.personalInfo.lastName}
              </Option>
            ))}
          </Select>
        </Form.Item>

        <Form.Item
          name="person2Id"
          label="Second Person"
          rules={[{ required: true, message: 'Please select second person' }]}
        >
          <Select placeholder="Select second person" showSearch optionFilterProp="children">
            {members.map((m: any) => (
              <Option key={m.id} value={m.id}>
                {m.personalInfo.firstName} {m.personalInfo.lastName}
              </Option>
            ))}
          </Select>
        </Form.Item>

        <Form.Item
          name="isCurrent"
          label="Current Relationship"
          valuePropName="checked"
        >
          <Switch />
        </Form.Item>

        <Form.Item>
          <Button type="primary" htmlType="submit" loading={loading} block>
            Create Relationship
          </Button>
        </Form.Item>
      </Form>
    </Modal>
  );
}

// src/features/relationships/pages/RelationshipsPage.tsx
import { useState } from 'react';
import { Table, Button, Tag, Space, Modal, message } from 'antd';
import { PlusOutlined, DeleteOutlined } from '@ant-design/icons';
import { useQuery, useMutation } from '@apollo/client';
import { GET_RELATIONSHIPS } from '../graphql/queries';
import { DELETE_RELATIONSHIP } from '../graphql/mutations';
import { RelationshipForm } from '../components/RelationshipForm';
import { useAuth } from '@/stores/auth';

export function RelationshipsPage() {
  const { user } = useAuth();
  const [formVisible, setFormVisible] = useState(false);

  const { loading, data } = useQuery(GET_RELATIONSHIPS, {
    variables: { familyId: user?.familyId },
    skip: !user?.familyId,
  });

  const [deleteRelationship] = useMutation(DELETE_RELATIONSHIP, {
    onCompleted: () => {
      message.success('Relationship deleted');
    },
    refetchQueries: ['GetRelationships'],
  });

  const handleDelete = (id: string) => {
    Modal.confirm({
      title: 'Delete Relationship',
      content: 'Are you sure you want to delete this relationship?',
      onOk: () => deleteRelationship({ variables: { id } }),
    });
  };

  const columns = [
    {
      title: 'Type',
      dataIndex: 'type',
      key: 'type',
      render: (type: string) => {
        const colors: Record<string, string> = {
          PARENT: 'blue',
          SPOUSE: 'pink',
          PARTNER: 'purple',
          ADOPTION: 'orange',
        };
        return <Tag color={colors[type]}>{type}</Tag>;
      },
    },
    {
      title: 'Person 1',
      dataIndex: 'person1Id',
      key: 'person1',
    },
    {
      title: 'Person 2',
      dataIndex: 'person2Id',
      key: 'person2',
    },
    {
      title: 'Current',
      dataIndex: 'isCurrent',
      key: 'isCurrent',
      render: (current: boolean) => (
        <Tag color={current ? 'green' : 'default'}>{current ? 'Yes' : 'No'}</Tag>
      ),
    },
    {
      title: 'Actions',
      key: 'actions',
      render: (_: any, record: any) => (
        <Button
          icon={<DeleteOutlined />}
          danger
          onClick={() => handleDelete(record.id)}
        />
      ),
    },
  ];

  return (
    <div className="p-6">
      <div className="flex justify-between items-center mb-6">
        <h1 className="text-2xl font-bold">Relationships</h1>
        <Button
          type="primary"
          icon={<PlusOutlined />}
          onClick={() => setFormVisible(true)}
        >
          Add Relationship
        </Button>
      </div>

      <Table
        columns={columns}
        dataSource={data?.relationships || []}
        rowKey="id"
        loading={loading}
      />

      <RelationshipForm
        visible={formVisible}
        onClose={() => setFormVisible(false)}
      />
    </div>
  );
}
