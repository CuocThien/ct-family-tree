import React, { useMemo, useRef, useEffect, useState } from 'react';
import * as d3 from 'd3';
import { Card } from 'antd';
import { TreeNode, TreeLink } from '../utils/layout';
import { TreeControls } from './TreeControls';

interface TreeChartProps {
  nodes: TreeNode[];
  links: TreeLink[];
  onNodeClick?: (node: TreeNode) => void;
}

export function TreeChart({ nodes, links, onNodeClick }: TreeChartProps) {
  const svgRef = useRef<SVGSVGElement>(null);
  const containerRef = useRef<HTMLDivElement>(null);
  const [zoom, setZoom] = useState(1);
  const [transform, setTransform] = useState({ x: 0, y: 0, k: 1 });

  const { width, height } = useMemo(() => {
    if (nodes.length === 0) return { width: 800, height: 600 };

    const xExtent = d3.extent(nodes, (d) => d.x) as [number, number];
    const yExtent = d3.extent(nodes, (d) => d.y) as [number, number];

    return {
      width: xExtent[1] - xExtent[0] + 400,
      height: yExtent[1] - yExtent[0] + 400,
    };
  }, [nodes]);

  useEffect(() => {
    if (!svgRef.current || nodes.length === 0) return;

    const svg = d3.select(svgRef.current);
    const g = svg.select<SVGGElement>('.tree-content');

    // Clear previous content
    g.selectAll('*').remove();

    // Calculate offset
    const xExtent = d3.extent(nodes, (d) => d.x) as [number, number];
    const yExtent = d3.extent(nodes, (d) => d.y) as [number, number];
    const offsetX = -xExtent[0] + 200;
    const offsetY = -yExtent[0] + 200;

    // Draw links
    const linkGroup = g.append('g').attr('class', 'links');

    links.forEach((link) => {
      const source = nodes.find((n) => n.id === link.source);
      const target = nodes.find((n) => n.id === link.target);

      if (source && target) {
        if (link.type === 'parent') {
          const path = d3.path();
          path.moveTo(source.x + offsetX + 90, source.y + offsetY + 80);
          path.bezierCurveTo(
            source.x + offsetX + 90,
            source.y + offsetY + 80 + (target.y + offsetY - source.y - offsetY - 80) / 2,
            target.x + offsetX + 90,
            source.y + offsetY + 80 + (target.y + offsetY - source.y - offsetY - 80) / 2,
            target.x + offsetX + 90,
            target.y + offsetY
          );
          linkGroup
            .append('path')
            .attr('d', path.toString())
            .attr('fill', 'none')
            .attr('stroke', '#d9d9d9')
            .attr('stroke-width', 2);
        } else {
          linkGroup
            .append('line')
            .attr('x1', source.x + offsetX + 180)
            .attr('y1', source.y + offsetY + 40)
            .attr('x2', target.x + offsetX)
            .attr('y2', target.y + offsetY + 40)
            .attr('stroke', '#ff7875')
            .attr('stroke-width', 2)
            .attr('stroke-dasharray', '5,5');
        }
      }
    });

    // Draw nodes
    const nodeGroup = g.append('g').attr('class', 'nodes');

    nodes.forEach((node) => {
      const nodeG = nodeGroup
        .append('g')
        .attr('transform', `translate(${node.x + offsetX}, ${node.y + offsetY})`)
        .attr('class', 'node')
        .style('cursor', 'pointer')
        .on('click', () => onNodeClick?.(node));

      // Card background
      nodeG
        .append('rect')
        .attr('width', 180)
        .attr('height', 80)
        .attr('rx', 8)
        .attr('fill', '#fff')
        .attr('stroke', node.gender === 'MALE' ? '#1890ff' : node.gender === 'FEMALE' ? '#eb2f96' : '#d9d9d9')
        .attr('stroke-width', 2);

      // Name
      nodeG
        .append('text')
        .attr('x', 90)
        .attr('y', 30)
        .attr('text-anchor', 'middle')
        .attr('font-size', '14px')
        .attr('font-weight', '600')
        .attr('fill', '#262626')
        .text(node.name.length > 20 ? node.name.slice(0, 20) + '...' : node.name);

      // Birth/Death info
      const infoText = node.birthYear
        ? `${node.birthYear}${node.deathYear ? ' - ' + node.deathYear : ''}`
        : '';
      if (infoText) {
        nodeG
          .append('text')
          .attr('x', 90)
          .attr('y', 50)
          .attr('text-anchor', 'middle')
          .attr('font-size', '12px')
          .attr('fill', '#8c8c8c')
          .text(infoText);
      }

      // Gender indicator
      if (node.gender) {
        const colors: Record<string, string> = {
          MALE: '#1890ff',
          FEMALE: '#eb2f96',
          OTHER: '#722ed1',
        };
        nodeG
          .append('circle')
          .attr('cx', 12)
          .attr('cy', 12)
          .attr('r', 6)
          .attr('fill', colors[node.gender] || '#d9d9d9');
      }
    });

    // Setup zoom behavior
    const zoomBehavior = d3.zoom<SVGSVGElement, unknown>()
      .scaleExtent([0.1, 3])
      .on('zoom', (event) => {
        g.attr('transform', event.transform);
        setTransform({
          x: event.transform.x,
          y: event.transform.y,
          k: event.transform.k,
        });
        setZoom(event.transform.k);
      });

    svg.call(zoomBehavior);

    // Initial zoom to fit
    if (containerRef.current) {
      const containerRect = containerRef.current.getBoundingClientRect();
      const scale = Math.min(
        containerRect.width / width,
        containerRect.height / height,
        1
      );
      const initialTransform = d3.zoomIdentity
        .translate(
          (containerRect.width - width * scale) / 2,
          (containerRect.height - height * scale) / 2
        )
        .scale(scale);
      svg.call(zoomBehavior.transform, initialTransform);
    }

  }, [nodes, links, onNodeClick, width, height]);

  const handleZoomIn = () => {
    if (svgRef.current) {
      d3.select(svgRef.current).transition().call(
        d3.zoom<SVGSVGElement, unknown>().scaleBy as any,
        1.3
      );
    }
  };

  const handleZoomOut = () => {
    if (svgRef.current) {
      d3.select(svgRef.current).transition().call(
        d3.zoom<SVGSVGElement, unknown>().scaleBy as any,
        0.7
      );
    }
  };

  const handleReset = () => {
    if (svgRef.current) {
      d3.select(svgRef.current).transition().call(
        d3.zoom<SVGSVGElement, unknown>().transform as any,
        d3.zoomIdentity
      );
    }
  };

  const handleFullscreen = () => {
    if (containerRef.current) {
      if (document.fullscreenElement) {
        document.exitFullscreen();
      } else {
        containerRef.current.requestFullscreen();
      }
    }
  };

  return (
    <div ref={containerRef} className="relative w-full h-[600px] border rounded-lg overflow-hidden bg-gray-50">
      <svg ref={svgRef} width="100%" height="100%">
        <g className="tree-content"></g>
      </svg>
      <TreeControls
        onZoomIn={handleZoomIn}
        onZoomOut={handleZoomOut}
        onReset={handleReset}
        onFullscreen={handleFullscreen}
        zoom={zoom}
      />
    </div>
  );
}

// src/features/tree/utils/ancestorLayout.ts
import { Member } from '@/types/graphql';
import { Relationship } from '@/types/graphql';

interface AncestorNode {
  id: string;
  member: Member;
  x: number;
  y: number;
  generation: number;
  father?: AncestorNode;
  mother?: AncestorNode;
}

export function buildAncestorTree(
  rootMemberId: string,
  members: Member[],
  relationships: Relationship[],
  maxGenerations: number = 4
): AncestorNode | null {
  const memberMap = new Map(members.map(m => [m.id, m]));

  // Build parent lookup
  const parentLookup = new Map<string, { fatherId?: string; motherId?: string }>();

  relationships.forEach(rel => {
    if (rel.type === 'PARENT') {
      const childId = rel.person2Id;
      const parentId = rel.person1Id;
      const parent = memberMap.get(parentId);

      if (!parentLookup.has(childId)) {
        parentLookup.set(childId, {});
      }

      const entry = parentLookup.get(childId)!;
      if (parent?.personalInfo.gender === 'MALE') {
        entry.fatherId = parentId;
      } else if (parent?.personalInfo.gender === 'FEMALE') {
        entry.motherId = parentId;
      } else {
        // If gender is not specified, assign to first available slot
        if (!entry.fatherId) {
          entry.fatherId = parentId;
        } else {
          entry.motherId = parentId;
        }
      }
    }
  });

  function buildNode(memberId: string, generation: number): AncestorNode | null {
    if (generation > maxGenerations) return null;

    const member = memberMap.get(memberId);
    if (!member) return null;

    const parents = parentLookup.get(memberId) || {};
    const node: AncestorNode = {
      id: memberId,
      member,
      x: 0,
      y: generation * 150,
      generation,
    };

    if (parents.fatherId) {
      node.father = buildNode(parents.fatherId, generation + 1);
    }
    if (parents.motherId) {
      node.mother = buildNode(parents.motherId, generation + 1);
    }

    return node;
  }

  function assignXPositions(node: AncestorNode, x: number, spread: number): number {
    node.x = x;

    let totalWidth = 0;
    const childSpread = spread / 2;

    if (node.father && node.mother) {
      totalWidth = assignXPositions(node.father, x - childSpread, childSpread);
      totalWidth += assignXPositions(node.mother, x + childSpread, childSpread);
    } else if (node.father) {
      totalWidth = assignXPositions(node.father, x - childSpread / 2, childSpread);
    } else if (node.mother) {
      totalWidth = assignXPositions(node.mother, x + childSpread / 2, childSpread);
    }

    return Math.max(spread, totalWidth);
  }

  const root = buildNode(rootMemberId, 0);
  if (root) {
    assignXPositions(root, 0, 800);
  }

  return root;
}

export function flattenAncestorTree(root: AncestorNode | null): AncestorNode[] {
  if (!root) return [];

  const result: AncestorNode[] = [];

  function traverse(node: AncestorNode) {
    result.push(node);
    if (node.father) traverse(node.father);
    if (node.mother) traverse(node.mother);
  }

  traverse(root);
  return result;
}

  relationships: Relationship[];

  maxGenerations?: number;
}

export function AncestorTree({
  rootMemberId,
  members,
  relationships,
  onNodeClick,
  maxGenerations = 4,
}: Props) {
  const svgRef = useRef<SVGSVGElement>(null);

  useEffect(() => {
    if (!svgRef.current || members.length === 0) return;

    // Build ancestor tree structure
    const memberMap = new Map(members.map(m => [m.id, m]));
    const parentLookup = new Map<string, { fatherId?: string; motherId?: string }>();

    relationships.forEach(rel => {
      if (rel.type === 'PARENT') {
        const childId = rel.person2Id;
        const parentId = rel.person1Id;
        const parent = memberMap.get(parentId);

        if (!parentLookup.has(childId)) {
          parentLookup.set(childId, {});
        }
        const entry = parentLookup.get(childId)!;
        if (parent?.personalInfo.gender === 'MALE') {
          entry.fatherId = parentId;
        } else if (parent?.personalInfo.gender === 'FEMALE') {
          entry.motherId = parentId;
        }
      }
    });

    // D3 hierarchy
    function buildHierarchy(memberId: string, depth: number): any {
      const member = memberMap.get(memberId);
      if (!member || depth > maxGenerations) return null;

      const parents = parentLookup.get(memberId) || {};
      const children: any[] = [];

      if (parents.fatherId) {
        const father = buildHierarchy(parents.fatherId, depth + 1);
        if (father) children.push({ ...father, side: 'father' });
      }
      if (parents.motherId) {
        const mother = buildHierarchy(parents.motherId, depth + 1);
        if (mother) children.push({ ...mother, side: 'mother' });
      }

      return {
        name: `${member.personalInfo.firstName} ${member.personalInfo.lastName}`,
        memberId,
        gender: member.personalInfo.gender,
        children: children.length > 0 ? children : undefined,
      };
    }

    const rootData = buildHierarchy(rootMemberId, 0);
    if (!rootData) return;

    // Create tree layout
    const svg = d3.select(svgRef.current);
    svg.selectAll('*').remove();

    const width = 1000;
    const height = 600;

    svg.attr('width', width).attr('height', height);

    const g = svg.append('g').attr('transform', 'translate(50, 50)');

    const treeLayout = d3.tree<any>().nodeSize([180, 120]);
    const root = d3.hierarchy(rootData);
    treeLayout(root);

    // Draw links
    g.selectAll('.link')
      .data(root.links())
      .join('path')
      .attr('class', 'link')
      .attr('fill', 'none')
      .attr('stroke', '#d9d9d9')
      .attr('stroke-width', 2)
      .attr('d', d3.linkVertical<any, any>()
        .x(d => d.x)
        .y(d => d.y)
      );

    // Draw nodes
    const nodes = g.selectAll('.node')
      .data(root.descendants())
      .join('g')
      .attr('class', 'node')
      .attr('transform', d => `translate(${d.x}, ${d.y})`)
      .style('cursor', 'pointer')
      .on('click', (_, d: any) => onNodeClick?.(d.data.memberId));

    nodes.append('rect')
      .attr('width', 140)
      .attr('height', 60)
      .attr('x', -70)
      .attr('y', -30)
      .attr('rx', 8)
      .attr('fill', '#fff')
      .attr('stroke', d => {
        const gender = d.data.gender;
        return gender === 'MALE' ? '#1890ff' : gender === 'FEMALE' ? '#eb2f96' : '#d9d9d9';
      })
      .attr('stroke-width', 2);

    nodes.append('text')
      .attr('dy', 5)
      .attr('text-anchor', 'middle')
      .attr('font-size', '12px')
      .attr('font-weight', '500')
      .text(d => d.data.name.length > 15 ? d.data.name.slice(0, 15) + '...' : d.data.name);

    // Fit to view
    const bounds = g.node()?.getBBox();
    if (bounds) {
      const scale = Math.min(
        (width - 100) / bounds.width,
        (height - 100) / bounds.height,
        1
      );
      g.attr('transform', `translate(${(width - bounds.width * scale) / 2}, ${(height - bounds.height * scale) / 2}) scale(${scale})`);
    }

  }, [rootMemberId, members, relationships, maxGenerations, onNodeClick]);

  return (
    <div className="border rounded-lg overflow-hidden bg-gray-50">
      <svg ref={svgRef} width="100%" height="600" />
    </div>
  );
}

import { Member, Relationship } from '@/types/graphql';

interface DescendantNode {
  id: string;
  member: Member;
  x: number;
  y: number;
  generation: number;
  children: DescendantNode[];
}

export function buildDescendantTree(
  rootMemberId: string,
  members: Member[],
  relationships: Relationship[],
  maxGenerations: number = 4
): DescendantNode | null {
  const memberMap = new Map(members.map(m => [m.id, m]));

  // Build children lookup
  const childrenLookup = new Map<string, string[]>();

  relationships.forEach(rel => {
    if (rel.type === 'PARENT') {
      const parentId = rel.person1Id;
      const childId = rel.person2Id;

      if (!childrenLookup.has(parentId)) {
        childrenLookup.set(parentId, []);
      }
      childrenLookup.get(parentId)!.push(childId);
    }
  });

  function buildNode(memberId: string, generation: number): DescendantNode | null {
    if (generation > maxGenerations) return null;

    const member = memberMap.get(memberId);
    if (!member) return null;

    // Get all children from both parents
    const childIds = childrenLookup.get(memberId) || [];
    const children: DescendantNode[] = [];

    childIds.forEach(childId => {
      const childNode = buildNode(childId, generation + 1);
      if (childNode) {
        children.push(childNode);
      }
    });

    return {
      id: memberId,
      member,
      x: 0,
      y: generation * 150,
      generation,
      children,
    };
  }

  function assignXPositions(node: DescendantNode): number {
    if (node.children.length === 0) {
      node.x = 0;
      return 180;
    }

    let totalWidth = 0;
    node.children.forEach((child, index) => {
      const childWidth = assignXPositions(child);
      child.x = totalWidth + childWidth / 2 - (node.children.length * 180) / 2;
      totalWidth += childWidth;
    });

    return Math.max(180, totalWidth);
  }

  const root = buildNode(rootMemberId, 0);
  if (root) {
    assignXPositions(root);
  }

  return root;
}

export function flattenDescendantTree(root: DescendantNode | null): DescendantNode[] {
  if (!root) return [];

  const result: DescendantNode[] = [root];
  root.children.forEach(child => {
    result.push(...flattenDescendantTree(child));
  });

  return result;
}

        const childId = rel.person2Id;
        if (!childrenLookup.has(parentId)) {
          childrenLookup.set(parentId, []);
        }
        childrenLookup.get(parentId)!.push(childId);
      }
    });

    function buildHierarchy(memberId: string): any {
      const member = memberMap.get(memberId);
      if (!member) return null;

      const childIds = childrenLookup.get(memberId) || [];
      const children = childIds
        .map(id => buildHierarchy(id))
        .filter(Boolean);

      return {
        name: `${member.personalInfo.firstName} ${member.personalInfo.lastName}`,
        memberId,
        gender: member.personalInfo.gender,
        children: children.length > 0 ? children : undefined,
      };
    }

    const rootData = buildHierarchy(rootMemberId);
    if (!rootData) return;

    const svg = d3.select(svgRef.current);
    svg.selectAll('*').remove();

    const width = 800;
    const height = 800;
    const radius = width / 2 - 100;

    svg.attr('width', width).attr('height', height);

    const g = svg.append('g')
      .attr('transform', `translate(${width / 2}, ${height / 2})`);

    // Create radial tree layout
    const tree = d3.tree<any>()
      .size([2 * Math.PI, radius])
      .separation((a, b) => (a.parent === b.parent ? 1 : 2) / a.depth);

    const root = d3.hierarchy(rootData);
    tree(root);

    // Draw links
    g.selectAll('.link')
      .data(root.links())
      .join('path')
      .attr('class', 'link')
      .attr('fill', 'none')
      .attr('stroke', '#d9d9d9')
      .attr('stroke-width', 1.5)
      .attr('d', d3.linkRadial<any, any>()
        .angle(d => d.x)
        .radius(d => d.y)
      );

    // Draw nodes
    const nodes = g.selectAll('.node')
      .data(root.descendants())
      .join('g')
      .attr('class', 'node')
      .attr('transform', d => `rotate(${d.x * 180 / Math.PI - 90}) translate(${d.y}, 0)`)
      .style('cursor', 'pointer')
      .on('click', (_, d: any) => onNodeClick?.(d.data.memberId));

    // Node circles
    nodes.append('circle')
      .attr('r', 6)
      .attr('fill', d => {
        const gender = d.data.gender;
        return gender === 'MALE' ? '#1890ff' : gender === 'FEMALE' ? '#eb2f96' : '#8c8c8c';
      })
      .attr('stroke', '#fff')
      .attr('stroke-width', 2);

    // Node labels
    nodes.append('text')
      .attr('dy', '0.31em')
      .attr('x', d => d.x < Math.PI === !d.children ? 12 : -12)
      .attr('text-anchor', d => d.x < Math.PI === !d.children ? 'start' : 'end')
      .attr('transform', d => d.x >= Math.PI ? 'rotate(180)' : null)
      .attr('font-size', '11px')
      .attr('fill', '#262626')
      .text(d => d.data.name.length > 12 ? d.data.name.slice(0, 12) + '...' : d.data.name);

  }, [rootMemberId, members, relationships, onNodeClick]);

  return (
    <div className="border rounded-lg overflow-hidden bg-gray-50 flex items-center justify-center">
      <svg ref={svgRef} width="800" height="800" />
    </div>
  );
}

// Update TreePage.tsx
import { useState, useMemo } from 'react';
import { useNavigate } from 'react-router-dom';
import { Spin, Empty, Select, message } from 'antd';
import { useQuery } from '@apollo/client';
import { TreeChart } from '../components/TreeChart';
import { AncestorTree } from '../components/AncestorTree';
import { DescendantTree } from '../components/DescendantTree';
import { RadialTree } from '../components/RadialTree';
import { TreeViewSelector, TreeViewType } from '../components/TreeViewSelector';
import { calculateTreeLayout } from '../utils/layout';
import { GET_MEMBERS } from '@/features/members/graphql/queries';
import { GET_RELATIONSHIPS } from '../graphql/queries';
import { useAuth } from '@/stores/auth';

export function TreePage() {
  const { user } = useAuth();
  const navigate = useNavigate();
  const [viewType, setViewType] = useState<TreeViewType>('standard');
  const [rootMemberId, setRootMemberId] = useState<string | undefined>();

  const { loading: loadingMembers, data: membersData } = useQuery(GET_MEMBERS, {
    variables: { familyId: user?.familyId },
    skip: !user?.familyId,
  });

  const { loading: loadingRels, data: relsData } = useQuery(GET_RELATIONSHIPS, {
    variables: { familyId: user?.familyId },
    skip: !user?.familyId,
  });

  const members = membersData?.members || [];
  const relationships = relsData?.relationships || [];

  // Set default root member
  useMemo(() => {
    if (!rootMemberId && members.length > 0) {
      // Find a member with no parents as default root
      const parentIds = new Set(
        relationships
          .filter((r: any) => r.type === 'PARENT')
          .map((r: any) => r.person2Id)
      );
      const rootMember = members.find((m: any) => !parentIds.has(m.id));
      setRootMemberId(rootMember?.id || members[0].id);
    }
  }, [members, relationships, rootMemberId]);

  const { nodes, links } = useMemo(() => {
    if (!members.length || !relationships.length) {
      return { nodes: [], links: [] };
    }
    return calculateTreeLayout(members, relationships);
  }, [members, relationships]);

  if (loadingMembers || loadingRels) {
    return (
      <div className="flex justify-center items-center h-64">
        <Spin size="large" />
      </div>
    );
  }

  if (members.length === 0) {
    return (
      <div className="p-6">
        <h1 className="text-2xl font-bold mb-6">Family Tree</h1>
        <Empty description="No family members to display" />
      </div>
    );
  }

  const handleNodeClick = (memberId: string) => {
    navigate(`/members/${memberId}`);
  };

  const renderTreeView = () => {
    switch (viewType) {
      case 'ancestor':
        return (
          <AncestorTree
            rootMemberId={rootMemberId || members[0].id}
            members={members}
            relationships={relationships}
            onNodeClick={handleNodeClick}
          />
        );
      case 'descendant':
        return (
          <DescendantTree
            rootMemberId={rootMemberId || members[0].id}
            members={members}
            relationships={relationships}
            onNodeClick={handleNodeClick}
          />
        );
      case 'radial':
        return (
          <RadialTree
            rootMemberId={rootMemberId || members[0].id}
            members={members}
            relationships={relationships}
            onNodeClick={handleNodeClick}
          />
        );
      default:
        return (
          <TreeChart
            nodes={nodes}
            links={links}
            onNodeClick={handleNodeClick}
          />
        );
    }
  };

  return (
    <div className="p-6">
      <h1 className="text-2xl font-bold mb-4">Family Tree</h1>

      <TreeViewSelector
        viewType={viewType}
        onViewTypeChange={setViewType}
        rootMemberId={rootMemberId}
        onRootMemberChange={setRootMemberId}
        members={members}
      />

      {renderTreeView()}
    </div>
  );
}
